## 4. 测试场景设计

### 4.1 场景1：并发登录测试

```yaml
# k6测试脚本 - 并发登录测试
import http from 'k6/http';
import { check, sleep } from 'k6';
import { Rate } from 'k6/metrics';

const errorRate = new Rate('errors');

export const options = {
  stages: [
    { duration: '30s', target: 100 },   // 预热
    { duration: '1m', target: 500 },    // 逐步增加
    { duration: '2m', target: 1000 },   // 峰值
    { duration: '1m', target: 1000 },   // 维持
    { duration: '30s', target: 0 },     // 冷却
  ],
  thresholds: {
    http_req_duration: ['p(95)<500', 'p(99)<1000'],
    errors: ['rate<0.01'],
  },
};

export default function () {
  const baseUrl = 'http://api-gateway:8888';
  
  // 1. 获取登录验证码
  const captchaRes = http.get(`${baseUrl}/api/v1/auth/captcha`);
  check(captchaRes, {
    'captcha status 200': (r) => r.status === 200,
  });
  
  const captchaData = JSON.parse(captchaRes.body);
  const captchaId = captchaData.data.captchaId;
  
  // 2. 用户登录
  const loginRes = http.post(`${baseUrl}/api/v1/auth/login`, 
    JSON.stringify({
      username: `user${__VU % 10000}`,
      password: 'password123',
      captchaId: captchaId,
      captchaCode: '1234',
    }),
    {
      headers: { 'Content-Type': 'application/json' },
    }
  );
  
  check(loginRes, {
    'login status 200': (r) => r.status === 200,
    'login has token': (r) => {
      const body = JSON.parse(r.body);
      return body.data && body.data.accessToken;
    },
  }) || errorRate.add(1);
  
  // 3. 验证token
  if (loginRes.status === 200) {
    const token = JSON.parse(loginRes.body).data.accessToken;
    
    const verifyRes = http.get(`${baseUrl}/api/v1/auth/verify`,
      {
        headers: { 'Authorization': `Bearer ${token}` },
      }
    );
    
    check(verifyRes, {
      'verify status 200': (r) => r.status === 200,
    });
  }
  
  sleep(1);
}
```

### 4.3 场景3：数据同步吞吐量测试

```bash
#!/bin/bash
# 数据同步吞吐量测试

# 测试配置
THREADS=(1 5 10 20 50 100)
RECORDS=100000

echo "=========================================="
echo "数据同步吞吐量测试"
echo "=========================================="

for threads in "${THREADS[@]}"; do
    echo ""
    echo "测试线程数: $threads"
    echo "----------------------------------------"
    
    # 执行测试
    START_TIME=$(date +%s)
    
    java -jar sync-test.jar \
        --threads $threads \
        --records $RECORDS \
        --target "http://api-gateway:8888/api/v1/import"
    
    END_TIME=$(date +%s)
    DURATION=$((END_TIME - START_TIME))
    
    # 计算指标
    THROUGHPUT=$(echo "scale=2; $RECORDS / $DURATION" | bc)
    
    echo "耗时: ${DURATION}秒"
    echo "吞吐量: ${THROUGHPUT} 条/秒"
done
```

### 4.4 场景4：缓存命中率测试

```java
// 缓存命中率测试
@Test
void testCacheHitRate() {
    // 准备测试数据
    List<Long> assetIds = generateAssetIds(1000);
    
    // 预热缓存
    for (Long assetId : assetIds.subList(0, 100)) {
        assetService.getById(assetId);  // 首次访问，缓存未命中
    }
    
    // 统计命中率
    AtomicInteger hitCount = new AtomicInteger(0);
    AtomicInteger missCount = new AtomicInteger(0);
    
    // 重复访问相同数据
    for (int i = 0; i < 1000; i++) {
        Long assetId = assetIds.get(i % 100);  // 访问前面100条
        
        // 监控缓存状态
        assetService.getById(assetId);
        
        // 检查是否命中缓存
        boolean hit = cacheManager.getCache("assets")
            .get(assetId.toString()) != null;
        
        if (hit) hitCount.incrementAndGet();
        else missCount.incrementAndGet();
    }
    
    // 计算命中率
    int total = hitCount.get() + missCount.get();
    double hitRate = (double) hitCount.get() / total * 100;
    
    System.out.println("缓存命中率: " + hitRate + "%");
    System.out.println("命中次数: " + hitCount.get());
    System.out.println("未命中次数: " + missCount.get());
    
    // 验证命中率
    assertThat(hitRate).isGreaterThan(90.0);
}
```
