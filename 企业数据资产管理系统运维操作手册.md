## 1. 日常运维操作

### 1.1 服务状态检查

```bash
#!/bin/bash
# scripts/daily-health-check.sh

echo "=========================================="
echo "企业数据资产管理系统 - 日常健康检查"
echo "检查时间: $(date '+%Y-%m-%d %H:%M:%S')"
echo "=========================================="

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# 检查计数器
PASS=0
FAIL=0
WARN=0

# 检查函数
check_service() {
    local name=$1
    local url=$2
    
    echo -n "[$name] "
    HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" $url --max-time 10)
    
    if [ "$HTTP_CODE" == "200" ]; then
        echo -e "${GREEN}✓ 正常 (HTTP $HTTP_CODE)${NC}"
        ((PASS++))
    elif [ "$HTTP_CODE" == "401" ]; then
        echo -e "${GREEN}✓ 正常 (HTTP $HTTP_CODE - 需认证)${NC}"
        ((PASS++))
    else
        echo -e "${RED}✗ 异常 (HTTP $HTTP_CODE)${NC}"
        ((FAIL++))
    fi
}

check_db() {
    local name=$1
    local host=$2
    local port=$3
    
    echo -n "[$name] "
    if nc -z -w5 $host $port 2>/dev/null; then
        echo -e "${GREEN}✓ 端口正常${NC}"
        ((PASS++))
    else
        echo -e "${RED}✗ 端口异常${NC}"
        ((FAIL++))
    fi
}

check_process() {
    local name=$1
    local process=$2
    
    echo -n "[$name] "
    if pgrep -f "$process" > /dev/null; then
        echo -e "${GREEN}✓ 进程运行中${NC}"
        ((PASS++))
    else
        echo -e "${RED}✗ 进程未运行${NC}"
        ((FAIL++))
    fi
}

# API网关健康检查
echo ""
echo ">>> API服务检查"
check_service "Gateway" "http://localhost:8888/actuator/health"
check_service "Auth服务" "http://localhost:8081/actuator/health"
check_service "Asset服务" "http://localhost:8082/actuator/health"
check_service "Lineage服务" "http://localhost:8083/actuator/health"
check_service "Quality服务" "http://localhost:8084/actuator/health"

# 数据库健康检查
echo ""
echo ">>> 数据库检查"
check_db "PostgreSQL" "localhost" "5432"
check_db "Redis" "localhost" "6379"
check_db "Neo4j" "localhost" "7687"
check_db "Elasticsearch" "localhost" "9200"
check_db "Kafka" "localhost" "9092"

# 中间件检查
echo ""
echo ">>> 中间件检查"
check_process "Kafka Broker" "kafka.Kafka"
check_process "Zookeeper" "org.apache.zookeeper"

# 资源使用检查
echo ""
echo ">>> 资源使用检查"
CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
MEM_USAGE=$(free | grep Mem | awk '{printf "%.2f", $3/$2 * 100}')
DISK_USAGE=$(df -h / | awk 'NR==2 {print $5}' | cut -d'%' -f1)

echo "CPU使用率: ${CPU_USAGE}%"
echo "内存使用率: ${MEM_USAGE}%"
echo "磁盘使用率: ${DISK_USAGE}%"

if (( $(echo "$CPU_USAGE > 80" | bc -l) )); then
    echo -e "${RED}⚠ CPU使用率过高${NC}"
    ((WARN++))
fi

if (( $(echo "$MEM_USAGE > 85" | bc -l) )); then
    echo -e "${RED}⚠ 内存使用率过高${NC}"
    ((WARN++))
fi

if [ "$DISK_USAGE" -gt 85 ]; then
    echo -e "${RED}⚠ 磁盘使用率过高${NC}"
    ((WARN++))
fi

# 日志检查
echo ""
echo ">>> 错误日志检查"
ERROR_COUNT=$(tail -n 1000 /var/log/edams/application.log 2>/dev/null | grep -c "ERROR" || echo "0")
if [ "$ERROR_COUNT" -gt 10 ]; then
    echo -e "${RED}⚠ 最近1000行日志中发现 $ERROR_COUNT 个ERROR${NC}"
    ((WARN++))
else
    echo -e "${GREEN}✓ 错误日志数量正常 ($ERROR_COUNT)${NC}"
fi

# 总结
echo ""
echo "=========================================="
echo "检查总结"
echo "=========================================="
echo -e "${GREEN}通过: $PASS${NC}"
echo -e "${YELLOW}警告: $WARN${NC}"
echo -e "${RED}失败: $FAIL${NC}"

if [ $FAIL -gt 0 ]; then
    echo ""
    echo -e "${RED}❌ 健康检查未通过，请检查失败项${NC}"
    exit 1
elif [ $WARN -gt 0 ]; then
    echo ""
    echo -e "${YELLOW}⚠ 健康检查有警告，请关注${NC}"
    exit 0
else
    echo ""
    echo -e "${GREEN}✅ 所有检查通过！${NC}"
    exit 0
fi
```

### 1.2 日志查看与分析

```bash
#!/bin/bash
# scripts/log-analysis.sh

# 日志路径
LOG_DIR="/var/log/edams"
APP_LOG="$LOG_DIR/application.log"
ACCESS_LOG="$LOG_DIR/access.log"
ERROR_LOG="$LOG_DIR/error.log"

# 颜色
RED='\033[0;31m'
BLUE='\033[0;34m'
NC='\033[0m'

show_usage() {
    echo "用法: $0 [选项]"
    echo ""
    echo "选项:"
    echo "  tail [n]        查看最近n行日志 (默认100)"
    echo "  error [n]      查看最近n条错误日志 (默认50)"
    echo "  search <关键字> 搜索包含关键字的日志"
    echo "  stats          显示日志统计信息"
    echo "  slow           查看慢查询日志"
    echo "  access [小时]   查看最近N小时的访问日志"
    echo ""
}

case "$1" in
    tail)
        LINES=${2:-100}
        tail -n $LINES $APP_LOG | \
            awk '{print strftime("%Y-%m-%d %H:%M:%S"), $0}' | \
            $0 --color
        ;;
        
    error)
        LINES=${2:-50}
        echo -e "${RED}=== 最近 $LINES 条错误日志 ===${NC}"
        grep "ERROR" $APP_LOG | tail -n $LINES | \
            awk '{print strftime("%Y-%m-%d %H:%M:%S"), $0}'
        ;;
        
    search)
        if [ -z "$2" ]; then
            echo "请提供搜索关键字"
            exit 1
        fi
        echo -e "${BLUE}=== 搜索结果: $2 ===${NC}"
        grep -i "$2" $APP_LOG | tail -n 100
        ;;
        
    stats)
        echo "=== 日志统计信息 ==="
        echo ""
        echo "总日志行数:"
        wc -l $APP_LOG
        
        echo ""
        echo "按级别统计:"
        grep -oE "\[LEVEL:[A-Z]+\]" $APP_LOG | \
            sort | uniq -c | sort -rn
        
        echo ""
        echo "每小时日志量:"
        grep -oE "[0-9]{4}-[0-9]{2}-[0-9]{2} [0-9]{2}:[0-9]{2}" $APP_LOG | \
            cut -d' ' -f1,2 | \
            awk -F'[ :]' '{print $1, $2":00"}' | \
            sort | uniq -c | sort -k2
        ;;
        
    slow)
        echo "=== 慢查询日志 (>1s) ==="
        grep -E "Slow Query|took: [0-9]+ms" $APP_LOG | tail -n 50
        ;;
        
    access)
        HOURS=${2:-1}
        SINCE=$(date -d "$HOURS hour ago" '+%Y-%m-%d %H:%M:%S')
        
        echo "=== 最近 $HOURS 小时访问统计 ==="
        awk -v since="$SINCE" '$0 ~ since {count++} END {print "总请求数:", count}' $ACCESS_LOG
        
        echo ""
        echo "状态码分布:"
        awk -v since="$SINCE" '$0 ~ since {print $9}' $ACCESS_LOG | \
            sort | uniq -c | sort -rn
        
        echo ""
        echo "慢请求 (>5s):"
        awk -v since="$SINCE" '$0 ~ since && $NF ~ /[5-9]\.[0-9]+$|1[0-9]+\.[0-9]+$/ {print $0}' $ACCESS_LOG | head
        ;;
        
    *)
        show_usage
        ;;
esac
```

### 2.1 常见故障处理流程

```markdown
## 故障处理流程

### 故障级别定义

| 级别 | 定义 | 响应时间 | 恢复时间 |
|------|------|----------|----------|
| P0 | 核心功能完全不可用 | 5分钟 | 1小时 |
| P1 | 核心功能部分不可用 | 15分钟 | 4小时 |
| P2 | 非核心功能不可用 | 1小时 | 24小时 |
| P3 | 功能异常/性能下降 | 4小时 | 72小时 |

### 故障处理流程

1. **故障发现**
   - 监控系统告警
   - 用户反馈
   - 主动巡检

2. **故障确认**
   - 确认故障现象
   - 评估影响范围
   - 确定故障级别

3. **故障通知**
   - 通知相关人员
   - 创建故障单
   - 启动应急响应

4. **故障定位**
   - 查看监控数据
   - 查看日志分析
   - 逐步排查

5. **故障恢复**
   - 执行应急预案
   - 验证服务状态
   - 确认数据完整性

6. **故障总结**
   - 编写故障报告
   - 分析根因
   - 制定改进措施
```

## 3. 备份恢复操作

### 3.1 数据库备份

```bash
#!/bin/bash
# scripts/backup.sh

# 配置
BACKUP_DIR="/data/backup"
DATE=$(date +%Y%m%d_%H%M%S)
RETENTION_DAYS=30

# 数据库配置
DB_HOST="localhost"
DB_PORT="5432"
DB_NAME="edams"
DB_USER="postgres"
DB_PASSWORD="password"

# S3配置
S3_BUCKET="s3://edams-backup/postgresql"
S3_ENDPOINT="http://localhost:9000"

# 创建备份目录
mkdir -p $BACKUP_DIR

echo "=========================================="
echo "数据库备份任务"
echo "开始时间: $(date)"
echo "=========================================="

# 1. PostgreSQL备份
echo ""
echo "[1/3] 备份 PostgreSQL..."
PG_BACKUP_FILE="$BACKUP_DIR/postgres_${DATE}.sql.gz"
PGPASSWORD=$DB_PASSWORD pg_dump -h $DB_HOST \
    -p $DB_PORT \
    -U $DB_USER \
    -d $DB_NAME \
    -Fc \
    -f $PG_BACKUP_FILE

if [ $? -eq 0 ]; then
    echo "✓ PostgreSQL备份成功: $PG_BACKUP_FILE"
    echo "  文件大小: $(du -h $PG_BACKUP_FILE | cut -f1)"
else
    echo "✗ PostgreSQL备份失败"
    exit 1
fi

# 2. Redis备份
echo ""
echo "[2/3] 备份 Redis..."
REDIS_BACKUP_FILE="$BACKUP_DIR/redis_${DATE}.rdb"
redis-cli SAVE
cp /var/lib/redis/dump.rdb $REDIS_BACKUP_FILE

if [ $? -eq 0 ]; then
    echo "✓ Redis备份成功: $REDIS_BACKUP_FILE"
else
    echo "✗ Redis备份失败"
fi

# 3. Neo4j备份
echo ""
echo "[3/3] 备份 Neo4j..."
NEO4J_BACKUP_FILE="$BACKUP_DIR/neo4j_${DATE}.tar.gz"
docker exec neo4j neo4j-admin database backup \
    --database=neo4j \
    --to-path=/backups/neo4j-latest
tar -czf $NEO4J_BACKUP_FILE -C /backups neo4j-latest

if [ $? -eq 0 ]; then
    echo "✓ Neo4j备份成功: $NEO4J_BACKUP_FILE"
else
    echo "✗ Neo4j备份失败"
fi

# 4. 上传到S3
echo ""
echo "[4/4] 上传到S3..."
mc cp $PG_BACKUP_FILE $S3_BUCKET/postgresql/
mc cp $REDIS_BACKUP_FILE $S3_BUCKET/redis/
mc cp $NEO4J_BACKUP_FILE $S3_BUCKET/neo4j/

if [ $? -eq 0 ]; then
    echo "✓ 上传成功"
else
    echo "⚠ 上传失败，请手动检查"
fi

# 5. 清理过期备份
echo ""
echo "[5/5] 清理过期备份..."
find $BACKUP_DIR -name "*.sql.gz" -mtime +$RETENTION_DAYS -delete
find $BACKUP_DIR -name "*.rdb" -mtime +$RETENTION_DAYS -delete
find $BACKUP_DIR -name "*.tar.gz" -mtime +$RETENTION_DAYS -delete

echo "✓ 清理完成"

# 总结
echo ""
echo "=========================================="
echo "备份完成"
echo "结束时间: $(date)"
echo "=========================================="
echo ""
echo "备份文件列表:"
ls -lh $BACKUP_DIR
```

### 3.2 数据恢复

```bash
#!/bin/bash
# scripts/restore.sh

set -e

# 配置
BACKUP_DIR="/data/backup"
DB_HOST="localhost"
DB_PORT="5432"
DB_NAME="edams"
DB_USER="postgres"

echo "=========================================="
echo "数据库恢复任务"
echo "=========================================="

# 列出可用备份
echo ""
echo "可用备份:"
ls -1 $BACKUP_DIR/*.sql.gz

read -p "请输入要恢复的备份文件名: " BACKUP_FILE

if [ ! -f "$BACKUP_DIR/$BACKUP_FILE" ]; then
    echo "错误: 备份文件不存在"
    exit 1
fi

# 确认恢复
echo ""
echo "⚠️ 警告：此操作将覆盖现有数据！"
read -p "确认恢复? (yes/no): " CONFIRM

if [ "$CONFIRM" != "yes" ]; then
    echo "取消恢复操作"
    exit 0
fi

# 停止应用服务
echo ""
echo "停止应用服务..."
kubectl scale deployment --all -n edams-prod --replicas=0

# 恢复PostgreSQL
echo ""
echo "恢复 PostgreSQL..."
PGPASSWORD=$DB_PASSWORD pg_restore -h $DB_HOST \
    -p $DB_PORT \
    -U $DB_USER \
    -d $DB_NAME \
    -c \
    -Fc \
    $BACKUP_DIR/$BACKUP_FILE

if [ $? -eq 0 ]; then
    echo "✓ PostgreSQL恢复成功"
else
    echo "✗ PostgreSQL恢复失败"
    exit 1
fi

# 重新启动应用服务
echo ""
echo "重新启动应用服务..."
kubectl scale deployment --all -n edams-prod --replicas=2

echo ""
echo "=========================================="
echo "恢复完成"
echo "=========================================="
```

## 4. 监控告警处理

### 4.1 告警级别与响应

```markdown
## 告警级别定义

| 级别 | 名称 | 描述 | 响应时间 | 通知方式 |
|------|------|------|----------|----------|
| P0 | 紧急 | 核心服务不可用 | 5分钟 | 电话+短信+邮件 |
| P1 | 严重 | 核心功能受损 | 15分钟 | 短信+邮件 |
| P2 | 警告 | 非核心功能异常 | 1小时 | 邮件 |
| P3 | 提示 | 轻微异常 | 4小时 | 邮件 |

## 告警处理流程

1. 收到告警 → 确认告警真实性
2. 评估影响 → 确定告警级别
3. 响应处理 → 按级别处理
4. 记录归档 → 记录处理过程
```

### 4.2 常见告警处理

```markdown
## 告警：服务响应时间 > 5s

**级别**: P1
**自动处理**: 发送通知
**手动处理**: 
1. 查看服务状态
2. 检查数据库连接
3. 检查缓存命中率
4. 如持续超过10分钟，执行扩容

## 告警：API错误率 > 1%

**级别**: P1
```

### 6.2 证书管理

```bash
#!/bin/bash
# scripts/ssl-cert-check.sh

DOMAINS=(
    "api.edams.com"
    "admin.edams.com"
    "sso.edams.com"
)

echo "=========================================="
echo "SSL证书检查"
echo "=========================================="

for domain in "${DOMAINS[@]}"; do
    echo ""
    echo "检查域名: $domain"
    
    # 获取证书信息
    CERT=$(echo | openssl s_client -servername $domain -connect $domain:443 2>/dev/null | \
            openssl x509 -noout -dates -subject 2>/dev/null)
    
    if [ -z "$CERT" ]; then
        echo "  ✗ 无法获取证书"
        continue
    fi
    
    # 解析证书
    echo "$CERT" | while read line; do
        echo "  $line"
    done
    
    # 检查过期时间
    EXPIRY_DATE=$(echo "$CERT" | grep notAfter | cut -d= -f2)
    DAYS_UNTIL=$(($(date -d "$EXPIRY_DATE" +%s) - $(date +%s) / 86400))
    
    echo "  剩余天数: $DAYS_UNTIL 天"
    
    if [ $DAYS_UNTIL -lt 30 ]; then
        echo "  ⚠️ 警告: 证书将在30天内过期"
    elif [ $DAYS_UNTIL -lt 0 ]; then
        echo "  ✗ 错误: 证书已过期"
    fi
done

echo ""
echo "=========================================="
```
