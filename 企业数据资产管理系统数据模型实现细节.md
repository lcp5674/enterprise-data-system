## 1. Neo4j图数据库实现

### 1.1 图模型设计

```cypher
// ============ 节点类型定义 ============

// 数据资产节点
(:DataAsset {
    id: Long,
    name: String,
    type: String,          // TABLE, VIEW, COLUMN, API, FILE, STREAM
    datasource_id: Long,
    schema: String,
    table_name: String,
    column_name: String,    // 列级血缘使用
    created_at: DateTime,
    updated_at: DateTime
})

// 数据源节点
(:DataSource {
    id: Long,
    name: String,
    type: String,          // MYSQL, POSTGRESQL, ORACLE, KAFKA, HIVE, etc.
    connection_url: String,
    created_at: DateTime
})

// 业务实体节点
(:BusinessEntity {
    id: Long,
    name: String,
    description: String,
    owner: String,
    domain: String         // 业务域
})

// 指标节点
(:Metric {
    id: Long,
    name: String,
    expression: String,
    aggregation: String,   // SUM, AVG, COUNT, etc.
    unit: String
})

// 标签节点
(:Tag {
    id: Long,
    name: String,
    category: String,
    color: String
})

// ============ 关系类型定义 ============

// 数据血缘关系
(:DataAsset)-[:DEPENDS_ON {
    type: String,          // ETL, SQL, VIEW, PROCEDURE, API, STREAM
    transformation: String, // 转换规则描述
    column_mapping: Map,   // 字段映射关系
    created_at: DateTime
}]->(:DataAsset)

// 归属关系
(:DataAsset)-[:BELONGS_TO]->(:DataSource)
(:DataAsset)-[:TAGGED_WITH]->(:Tag)
(:DataAsset)-[:REALIZES]->(:BusinessEntity)

// 指标计算依赖
(:Metric)-[:CALCULATED_FROM]->(:DataAsset)
(:Metric)-[:AGGREGATES]->(:Metric)

// 引用关系
(:DataAsset)-[:REFERENCES]->(:BusinessEntity)
```

### 1.2 Cypher查询示例

```cypher
// ============ 基础查询 ============

// 1. 查询资产基本信息
MATCH (a:DataAsset {id: $assetId})
RETURN a

// 2. 查询资产的上游血缘（数据来源）
MATCH (upstream:DataAsset)-[r:DEPENDS_ON]->(a:DataAsset {id: $assetId})
RETURN upstream, r

// 3. 查询资产的下游影响（数据去向）
MATCH (a:DataAsset {id: $assetId})-[r:DEPENDS_ON]->(downstream:DataAsset)
RETURN downstream, r

// 4. 查询完整血缘链路（递归）
MATCH path = (a:DataAsset {id: $assetId})-[:DEPENDS_ON*]->(end:DataAsset)
WHERE NOT (end)-[:DEPENDS_ON]->()
RETURN path

// 5. 查询指定深度的血缘
MATCH path = (a:DataAsset {id: $assetId})-[:DEPENDS_ON*1..3]->(related:DataAsset)
RETURN path

// ============ 血缘分析查询 ============

// 1. 检测循环依赖
MATCH path = (a:DataAsset)-[:DEPENDS_ON+]->(a)
WHERE a.id = $assetId
RETURN path AS circularPath

// 2. 影响分析：找出所有受变更影响的资产
MATCH (changed:DataAsset)-[:DEPENDS_ON*0..]->(affected:DataAsset)
WHERE changed.id = $changedAssetId
RETURN DISTINCT affected

// 3. 血缘追溯：从目标资产追溯所有数据来源
MATCH (root:DataAsset)-[:DEPENDS_ON*]->(target:DataAsset {id: $targetAssetId})
WHERE NOT (root)-[:DEPENDS_ON]->()
RETURN root

// 4. 跨系统血缘查询
MATCH (a:DataAsset {datasource_id: $datasource1})-[:DEPENDS_ON*]->(b:DataAsset {datasource_id: $datasource2})
WHERE a.id = $startAssetId
RETURN b

// ============ 质量血缘分析 ============

// 1. 查询资产质量分数与血缘关系
MATCH (a:DataAsset)
OPTIONAL MATCH (a)-[:DEPENDS_ON]->(upstream:DataAsset)
WITH a, COLLECT(DISTINCT upstream.quality_score) AS upstreamScores
WITH a, upstreamScores,
     CASE 
         WHEN size(upstreamScores) = 0 THEN a.quality_score
         ELSE REDUCE(sum = 0.0, score IN upstreamScores | sum + score) / size(upstreamScores)
     END AS avgUpstreamScore
RETURN a, avgUpstreamScore

// 2. 质量影响链路分析
MATCH (lowQuality:DataAsset)-[:DEPENDS_ON*]->(affected:DataAsset)
WHERE lowQuality.quality_score < 60
RETURN lowQuality, affected, LENGTH((lowQuality)-[:DEPENDS_ON*]->(affected)) AS depth

// ============ 业务语义血缘 ============

// 1. 业务实体到数据资产的映射
MATCH (be:BusinessEntity)-[:REALIZES]->(asset:DataAsset)
RETURN be, COLLECT(asset) AS assets

// 2. 指标到数据资产的追溯
MATCH (m:Metric)-[:CALCULATED_FROM]->(a:DataAsset)
RETURN m, COLLECT(a) AS sourceAssets

// 3. 标签关联分析
MATCH (a:DataAsset)-[:TAGGED_WITH]->(t:Tag)
WITH t, COLLECT(a) AS taggedAssets, COUNT(a) AS assetCount
WHERE assetCount > 10
RETURN t, assetCount, taggedAssets

// ============ 性能优化查询 ============

// 1. 使用索引的查询
// 确保索引存在
CREATE INDEX asset_id IF NOT EXISTS FOR (a:DataAsset) ON (a.id);
CREATE INDEX asset_name IF NOT EXISTS FOR (a:DataAsset) ON (a.name);
CREATE INDEX asset_type IF NOT EXISTS FOR (a:DataAsset) ON (a.type);
CREATE INDEX asset_datasource IF NOT EXISTS FOR (a:DataAsset) ON (a.datasource_id);

// 2. 查询优化：限制返回量
MATCH (a:DataAsset)-[:DEPENDS_ON]->(b:DataAsset)
WHERE a.datasource_id = $datasourceId
RETURN a, b
LIMIT 1000

// 3. 使用PROFILE分析查询
PROFILE 
MATCH path = (a:DataAsset {id: $assetId})-[:DEPENDS_ON*1..5]->(end:DataAsset)
RETURN path
```

### 1.3 Java集成代码

```java
package com.enterprise.edams.lineage.repository;

import com.enterprise.edams.lineage.entity.GraphNode;
import com.enterprise.edams.lineage.entity.GraphRelation;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.neo4j.driver.*;
import org.springframework.stereotype.Repository;

import java.util.*;

/**
 * Neo4j血缘图谱仓储
 */
@Slf4j
@Repository
@RequiredArgsConstructor
public class LineageGraphRepository {

    private final Driver driver;
    
    // ============ 节点操作 ============
    
    /**
     * 创建数据资产节点
     */
    public void createAssetNode(Long assetId, String name, String type, Long datasourceId) {
        String cypher = """
            MERGE (a:DataAsset {id: $assetId})
            SET a.name = $name,
                a.type = $type,
                a.datasource_id = $datasourceId,
                a.updated_at = datetime()
            RETURN a
            """;
        
        try (Session session = driver.session()) {
            session.executeWrite(tx -> tx.run(cypher, 
                Values.parameters(
                    "assetId", assetId,
                    "name", name,
                    "type", type,
                    "datasourceId", datasourceId
                )
            ).consume());
        }
    }
    
    /**
     * 批量创建节点
     */
    public void batchCreateAssetNodes(List<GraphNode> nodes) {
        String cypher = """
            UNWIND $nodes AS node
            MERGE (a:DataAsset {id: node.id})
            SET a.name = node.name,
                a.type = node.type,
                a.datasource_id = node.datasourceId,
                a.updated_at = datetime()
            """;
        
        try (Session session = driver.session()) {
            Map<String, Object>[] nodeMaps = nodes.stream()
                .map(n -> Map.of(
                    "id", n.getId(),
                    "name", n.getName(),
                    "type", n.getType(),
                    "datasourceId", n.getDatasourceId()
                ))
                .toArray(Map[]::new);
            
            session.executeWrite(tx -> tx.run(cypher, 
                Values.parameters("nodes", nodeMaps)
            ).consume());
        }
    }
    
    // ============ 关系操作 ============
    
    /**
     * 创建血缘关系
     */
    public void createLineageRelation(Long sourceId, Long targetId, 
                                       String relationType, Map<String, Object> properties) {
        String cypher = """
            MATCH (source:DataAsset {id: $sourceId})
            MATCH (target:DataAsset {id: $targetId})
            MERGE (source)-[r:DEPENDS_ON]->(target)
            SET r.type = $relationType,
                r.transformation = $transformation,
                r.column_mapping = $columnMapping,
                r.updated_at = datetime()
            """;
        
        try (Session session = driver.session()) {
            session.executeWrite(tx -> tx.run(cypher,
                Values.parameters(
                    "sourceId", sourceId,
                    "targetId", targetId,
                    "relationType", relationType,
                    "transformation", properties.getOrDefault("transformation", ""),
                    "columnMapping", properties.getOrDefault("columnMapping", Map.of())
                )
            ).consume());
        }
    }
    
    /**
     * 删除血缘关系
     */
    public void deleteLineageRelation(Long sourceId, Long targetId) {
        String cypher = """
            MATCH (source:DataAsset {id: $sourceId})-[r:DEPENDS_ON]->(target:DataAsset {id: $targetId})
            DELETE r
            """;
        
        try (Session session = driver.session()) {
            session.executeWrite(tx -> tx.run(cypher,
                Values.parameters("sourceId", sourceId, "targetId", targetId)
            ).consume());
        }
    }
    
    // ============ 查询操作 ============
    
    /**
     * 获取上游血缘（数据来源）
     */
    public List<GraphNode> getUpstreamLineage(Long assetId, int depth) {
        String cypher = """
            MATCH path = (upstream:DataAsset)-[:DEPENDS_ON*1..%d]->(target:DataAsset {id: $assetId})
            RETURN upstream, LENGTH(path) AS distance
            ORDER BY distance
            """.formatted(depth);
        
        try (Session session = driver.session()) {
            Result result = session.executeRead(tx -> tx.run(cypher,
                Values.parameters("assetId", assetId)
            ));
            
            List<GraphNode> nodes = new ArrayList<>();
            while (result.hasNext()) {
                Record record = result.next();
                Node node = record.get("upstream").asNode();
                nodes.add(toGraphNode(node));
            }
            return nodes;
        }
    }
    
    /**
     * 获取下游血缘（数据去向）
     */
    public List<GraphNode> getDownstreamLineage(Long assetId, int depth) {
        String cypher = """
            MATCH path = (source:DataAsset {id: $assetId})-[:DEPENDS_ON*1..%d]->(downstream:DataAsset)
            RETURN downstream, LENGTH(path) AS distance
            ORDER BY distance
            """.formatted(depth);
        
        try (Session session = driver.session()) {
            Result result = session.executeRead(tx -> tx.run(cypher,
                Values.parameters("assetId", assetId)
            ));
            
            List<GraphNode> nodes = new ArrayList<>();
            while (result.hasNext()) {
                Record record = result.next();
                Node node = record.get("downstream").asNode();
                nodes.add(toGraphNode(node));
            }
            return nodes;
        }
    }
    
    /**
     * 获取完整血缘图
     */
    public Map<String, Object> getFullLineageGraph(Long assetId, int maxDepth) {
        String cypher = """
            MATCH path = (a:DataAsset {id: $assetId})-[:DEPENDS_ON*0..%d]-(related:DataAsset)
            RETURN 
                COLLECT(DISTINCT a) AS nodes,
                COLLECT(DISTINCT relationships(path)) AS rels
            """.formatted(maxDepth);
        
        try (Session session = driver.session()) {
            Result result = session.executeRead(tx -> tx.run(cypher,
                Values.parameters("assetId", assetId)
            ));
            
            if (result.hasNext()) {
                Record record = result.next();
                List<Node> nodeList = record.get("nodes").asList(Value::asNode);
                List<List<Relationship>> relList = record.get("rels").asList(v -> v.asList(Value::asRelationship));
                
                return Map.of(
                    "nodes", nodeList.stream().map(this::toGraphNode).toList(),
                    "edges", relList.stream().flatMap(List::stream)
                        .map(this::toGraphEdge).toList()
                );
            }
            return Map.of("nodes", List.of(), "edges", List.of());
        }
    }
    
    /**
     * 检测循环依赖
     */
    public boolean hasCircularDependency(Long assetId) {
        String cypher = """
            MATCH path = (a:DataAsset {id: $assetId})-[:DEPENDS_ON+]->(a)
            RETURN path
            LIMIT 1
            """;
        
        try (Session session = driver.session()) {
            Result result = session.executeRead(tx -> tx.run(cypher,
                Values.parameters("assetId", assetId)
            ));
            return result.hasNext();
        }
    }
    
    /**
     * 影响分析：找出所有受影响的资产
     */
    public List<GraphNode> analyzeImpact(Long assetId) {
        String cypher = """
            MATCH (changed:DataAsset)-[:DEPENDS_ON*0..]->(affected:DataAsset)
            WHERE changed.id = $assetId
            RETURN DISTINCT affected
            """;
        
        try (Session session = driver.session()) {
            Result result = session.executeRead(tx -> tx.run(cypher,
                Values.parameters("assetId", assetId)
            ));
            
            List<GraphNode> affected = new ArrayList<>();
            while (result.hasNext()) {
                Node node = result.next().get("affected").asNode();
                affected.add(toGraphNode(node));
            }
            return affected;
        }
    }
    
    // ============ 批量操作 ============
    
    /**
     * 批量导入血缘关系
     */
    public void batchImportLineage(List<GraphRelation> relations) {
        String cypher = """
            UNWIND $relations AS rel
            MATCH (source:DataAsset {id: rel.sourceId})
            MATCH (target:DataAsset {id: rel.targetId})
            MERGE (source)-[r:DEPENDS_ON]->(target)
            SET r.type = rel.type,
                r.transformation = rel.transformation,
                r.column_mapping = rel.columnMapping,
                r.updated_at = datetime()
            """;
        
        try (Session session = driver.session()) {
            Map<String, Object>[] relMaps = relations.stream()
                .map(r -> Map.of(
                    "sourceId", r.getSourceId(),
                    "targetId", r.getTargetId(),
                    "type", r.getType(),
                    "transformation", r.getTransformation(),
                    "columnMapping", r.getColumnMapping()
                ))
                .toArray(Map[]::new);
            
            session.executeWrite(tx -> tx.run(cypher,
                Values.parameters("relations", relMaps)
            ).consume());
        }
    }
    
    // ============ 工具方法 ============
    
    private GraphNode toGraphNode(Node node) {
        return GraphNode.builder()
            .id(node.id())
            .properties(Map.copyOf(node.asMap()))
            .labels(List.copyOf(node.labels()))
            .build();
    }
    
    private GraphEdge toGraphEdge(Relationship rel) {
        return GraphEdge.builder()
            .id(rel.id())
            .sourceId((Long) rel.startNodeId())
            .targetId((Long) rel.endNodeId())
            .type(rel.type())
            .properties(Map.copyOf(rel.asMap()))
            .build();
    }
}
```

```cypher
// 单属性索引
CREATE INDEX asset_id IF NOT EXISTS FOR (a:DataAsset) ON (a.id);
CREATE INDEX asset_name IF NOT EXISTS FOR (a:DataAsset) ON (a.name);
CREATE INDEX asset_type IF NOT EXISTS FOR (a:DataAsset) ON (a.type);
CREATE INDEX asset_datasource IF NOT EXISTS FOR (a:DataAsset) ON (a.datasource_id);

// 复合索引

// 文本索引（用于模糊搜索）
CREATE INDEX asset_name_text IF NOT EXISTS 
FOR (a:DataAsset) ON (a.name)
OPTIONS {indexProvider: 'text-1.0'};

// ============ 关系索引 ============

// 为关系类型创建索引
CREATE INDEX lineage_type IF NOT EXISTS 
FOR ()-[r:DEPENDS_ON]->() ON (r.type);

// ============ 约束 ============

// 唯一性约束
CREATE CONSTRAINT asset_id_unique IF NOT EXISTS
FOR (a:DataAsset) REQUIRE a.id IS UNIQUE;

CREATE CONSTRAINT datasource_id_unique IF NOT EXISTS
FOR (d:DataSource) REQUIRE d.id IS UNIQUE;

CREATE CONSTRAINT tag_name_unique IF NOT EXISTS
FOR (t:Tag) REQUIRE (t.name, t.category) IS UNIQUE;
```

```java
package com.enterprise.edams.search.repository;

import com.enterprise.edams.search.dto.*;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageImpl;
import org.springframework.data.domain.Sort;
import org.springframework.data.elasticsearch.client.elc.ElasticsearchTemplate;
import org.springframework.data.elasticsearch.core.SearchHit;
import org.springframework.data.elasticsearch.core.SearchHits;
import org.springframework.data.elasticsearch.core.query.Criteria;
import org.springframework.data.elasticsearch.core.query.CriteriaQuery;
import org.springframework.data.elasticsearch.core.query.Query;
import org.springframework.stereotype.Repository;

import java.util.List;
import java.util.stream.Collectors;

/**
 * Elasticsearch资产管理仓储
 */
@Slf4j
@Repository
@RequiredArgsConstructor
public class AssetSearchRepository {

    private final ElasticsearchTemplate elasticsearchTemplate;

    // ============ 索引操作 ============

    /**
     * 创建索引
     */
    public void createIndex(String indexName) {
        if (!elasticsearchTemplate.indexOps(IndexCoordinates.of(indexName)).exists()) {
            elasticsearchTemplate.indexOps(IndexCoordinates.of(indexName))
                .create();
            log.info("索引创建成功: {}", indexName);
        }
    }

    /**
     * 删除索引
     */
    public void deleteIndex(String indexName) {
        if (elasticsearchTemplate.indexOps(IndexCoordinates.of(indexName)).exists()) {
            elasticsearchTemplate.indexOps(IndexCoordinates.of(indexName))
                .delete();
            log.info("索引删除成功: {}", indexName);
        }
    }

    // ============ 文档操作 ============

    /**
     * 索引单个文档
     */
    public void indexDocument(AssetDocument document) {
        elasticsearchTemplate.save(document);
        log.debug("文档索引成功: id={}", document.getId());
    }

    /**
     * 批量索引文档
     */
    public void bulkIndex(List<AssetDocument> documents) {
        if (documents.isEmpty()) {
            return;
        }
        elasticsearchTemplate.save(documents);
        log.info("批量索引完成: count={}", documents.size());
    }

    /**
     * 删除文档
     */
    public void deleteDocument(Long assetId) {
        elasticsearchTemplate.delete(String.valueOf(assetId), AssetDocument.class);
        log.debug("文档删除成功: id={}", assetId);
    }

    // ============ 查询操作 ============

    /**
     * 全文搜索
     */
    public Page<AssetDocument> search(SearchRequest request) {
        Criteria criteria = new Criteria();
        
        // 关键词搜索
        if (request.getKeyword() != null && !request.getKeyword().isBlank()) {
            criteria = criteria.and(new Criteria("search_content")
                .matches(request.getKeyword()));
        }
        
        // 类型过滤
        if (request.getAssetTypes() != null && !request.getAssetTypes().isEmpty()) {
            criteria = criteria.and(new Criteria("asset_type")
                .in(request.getAssetTypes()));
        }
        
        // 敏感级别过滤
        if (request.getSensitivities() != null && !request.getSensitivities().isEmpty()) {
            criteria = criteria.and(new Criteria("sensitivity")
                .in(request.getSensitivities()));
        }
        
        // 状态过滤
        if (request.getStatuses() != null && !request.getStatuses().isEmpty()) {
            criteria = criteria.and(new Criteria("status")
                .in(request.getStatuses()));
        }
        
        // 数据源过滤
        if (request.getDatasourceId() != null) {
            criteria = criteria.and(new Criteria("datasource_id")
                .is(request.getDatasourceId()));
        }
        
        // 所有者过滤
        if (request.getOwnerId() != null) {
            criteria = criteria.and(new Criteria("owner_id")
                .is(request.getOwnerId()));
        }
        
        // 标签过滤
        if (request.getTags() != null && !request.getTags().isEmpty()) {
            criteria = criteria.and(new Criteria("tags")
                .in(request.getTags()));
        }
        
        // 质量分数范围
        if (request.getMinQualityScore() != null || request.getMaxQualityScore() != null) {
            Criteria qualityCriteria = new Criteria("quality_score");
            if (request.getMinQualityScore() != null) {
                qualityCriteria = qualityCriteria.greaterThanEqual(request.getMinQualityScore());
            }
            if (request.getMaxQualityScore() != null) {
                qualityCriteria = qualityCriteria.lessThanEqual(request.getMaxQualityScore());
            }
            criteria = criteria.and(qualityCriteria);
        }

        // 构建查询
        Query query = new CriteriaQuery(criteria)
            .setPageable(request.toPageable());

        // 执行查询
        SearchHits<AssetDocument> searchHits = elasticsearchTemplate.search(
            query, AssetDocument.class);

        // 转换结果
        List<AssetDocument> documents = searchHits.getSearchHits().stream()
            .map(SearchHit::getContent)
            .collect(Collectors.toList());

        return new PageImpl<>(documents, request.toPageable(), searchHits.getTotalHits());
    }

    /**
     * 自动补全搜索
     */
    public List<String> autocomplete(String prefix, int limit) {
        Criteria criteria = new Criteria("asset_name.autocomplete")
            .startsWith(prefix);

        Query query = new CriteriaQuery(criteria)
            .setPageable(org.springframework.data.domain.PageRequest.of(0, limit))
            .setSourceFilter(new org.springframework.data.elasticsearch.core.query.FetchSourceFilter(
                new String[]{"asset_name"}, null));

        SearchHits<AssetDocument> searchHits = elasticsearchTemplate.search(
            query, AssetDocument.class);

        return searchHits.getSearchHits().stream()
            .map(hit -> hit.getContent().getAssetName())
            .distinct()
            .collect(Collectors.toList());
    }

    /**
     * 高亮搜索
     */
    public Page<AssetDocument> searchWithHighlight(SearchRequest request) {
        Criteria criteria = new Criteria("search_content")
            .matches(request.getKeyword());

        // 添加过滤条件
        if (request.getAssetTypes() != null) {
            criteria = criteria.and(new Criteria("asset_type").in(request.getAssetTypes()));
        }

        Query query = new CriteriaQuery(criteria)
            .setPageable(request.toPageable());

        // 设置高亮
        query.addHighlightQuery(new HighlightQuery(
            new Highlight.Field("asset_name")
                .withPreTags("<em>")
                .withPostTags("</em>"),
            AssetDocument.class
        ));
        query.addHighlightQuery(new HighlightQuery(
            new Highlight.Field("description")
                .withPreTags("<em>")
                .withPostTags("</em>"),
            AssetDocument.class
        ));

        SearchHits<AssetDocument> searchHits = elasticsearchTemplate.search(
            query, AssetDocument.class);

        List<AssetDocument> documents = searchHits.getSearchHits().stream()
            .map(hit -> {
                AssetDocument doc = hit.getContent();
                // 可以在这里处理高亮片段
                // doc.setHighlightFields(hit.getHighlightFields());
                return doc;
            })
            .collect(Collectors.toList());

        return new PageImpl<>(documents, request.toPageable(), searchHits.getTotalHits());
    }

    /**
     * 聚合统计
     */
    public SearchAggregation aggregate(String field) {
        NativeQuery query = NativeQuery.builder()
            .withAggregation(AggregationBuilders.terms(field).field(field).size(100))
            .withAggregation(AggregationBuilders.count("total").field("id"))
            .withMaxResults(0)
            .build();

        SearchHits<AssetDocument> searchHits = elasticsearchTemplate.search(query, AssetDocument.class);

        Map<String, Long> buckets = new HashMap<>();
        searchHits.getAggregations().getAsMap().forEach((name, agg) -> {
            if (agg instanceof StringTerms) {
                StringTerms terms = (StringTerms) agg;
                terms.getBuckets().forEach(bucket -> 
                    buckets.put(bucket.getKey().stringValue(), bucket.getDocCount()));
            }
        });

        return new SearchAggregation(buckets);
    }

        List<AssetDocument> documents = searchHits.getSearchHits().stream()
            .map(SearchHit::getContent)
            .collect(Collectors.toList());

        return new PageImpl<>(documents);
```

```java
package com.enterprise.edams.search.service;

import com.enterprise.edams.search.dto.AssetDocument;
import lombok.RequiredArgsConstructor;

/**
 * ES数据同步服务

@Slf4j
@Service
@RequiredArgsConstructor
public class ElasticsearchSyncService {

    private final AssetSearchRepository searchRepository;
    private final AssetRepository assetRepository;
    private final RedisTemplate<String, Object> redisTemplate;

    private static final String SYNC_CACHE_KEY = "es:sync:version";

    // ============ 消息队列同步 ============

    /**
     * 监听资产变更消息
     */
    @KafkaListener(topics = "asset-changes", groupId = "es-sync-group")
    public void handleAssetChange(ConsumerRecord<String, AssetChangeEvent> record) {
        AssetChangeEvent event = record.value();
        log.info("收到资产变更事件: type={}, id={}", event.getType(), event.getAssetId());

        try {
            switch (event.getType()) {
                case CREATE:
                case UPDATE:
                    syncAssetToES(event.getAssetId());
                    break;
                case DELETE:
                    searchRepository.deleteDocument(event.getAssetId());
                    break;
            }
            
            // 更新同步版本号
            updateSyncVersion();
            
        } catch (Exception e) {
            log.error("ES同步失败: id={}", event.getAssetId(), e);
            // 发送到死信队列
            sendToDLQ(event);
        }
    }

    /**
     * 同步单个资产到ES
     */
    private void syncAssetToES(Long assetId) {
        Asset asset = assetRepository.findById(assetId)
            .orElseThrow(() -> new AssetNotFoundException(assetId));
        
        AssetDocument document = toDocument(asset);
        searchRepository.indexDocument(document);
        
        log.debug("资产同步到ES成功: id={}", assetId);
    }

    // ============ 批量全量同步 ============

    /**
     * 全量同步
     */
    public void fullSync() {
        log.info("开始全量同步ES数据...");
        
        long startTime = System.currentTimeMillis();
        int batchSize = 1000;
        int offset = 0;
        int totalSynced = 0;

        while (true) {
            List<Asset> assets = assetRepository.findAll(offset, batchSize);
            if (assets.isEmpty()) {
                break;
            }

            List<AssetDocument> documents = assets.stream()
                .map(this::toDocument)
                .collect(Collectors.toList());

            searchRepository.bulkIndex(documents);
            
            totalSynced += assets.size();
            offset += batchSize;
            
            log.info("全量同步进度: {}/{}", totalSynced, assetRepository.count());
        }

        long duration = System.currentTimeMillis() - startTime;
        log.info("全量同步完成: total={}, duration={}ms", totalSynced, duration);
    }

    /**
     * 增量同步
     */
    public void incrementalSync() {
        log.info("开始增量同步ES数据...");
        
        Long lastSyncVersion = getLastSyncVersion();
        List<Asset> changedAssets = assetRepository.findChangedAfter(lastSyncVersion);
        
        if (changedAssets.isEmpty()) {
            log.info("没有需要同步的数据");
            return;
        }

        List<AssetDocument> documents = changedAssets.stream()
            .map(this::toDocument)
            .collect(Collectors.toList());

        searchRepository.bulkIndex(documents);
        
        updateSyncVersion();
        
        log.info("增量同步完成: count={}", changedAssets.size());
    }

    // ============ 定时同步 ============

    /**
     * 定时全量同步任务
     */
    @Scheduled(cron = "0 0 2 * * ?")  // 每天凌晨2点
    public void scheduledFullSync() {
        log.info("定时全量同步任务开始");
        try {
            // 重建索引
            String indexName = "data_assets_" + System.currentTimeMillis();
            searchRepository.createIndex(indexName);
            
            // 切换索引
            switchAlias(indexName);
            
            // 全量同步
            fullSync();
            
        } catch (Exception e) {
            log.error("定时全量同步失败", e);
        }
    }

    // ============ 工具方法 ============

    private AssetDocument toDocument(Asset asset) {
        return AssetDocument.builder()
            .id(asset.getId())
            .assetName(asset.getAssetName())
            .assetType(asset.getAssetType().name())
            .description(asset.getDescription())
            .ownerId(asset.getOwnerId())
            .ownerName(asset.getOwnerName())
            .sensitivity(asset.getSensitivity().name())
            .status(asset.getStatus().name())
            .datasourceId(asset.getDatasourceId())
            .datasourceName(asset.getDatasourceName())
            .databaseName(asset.getDatabaseName())
            .schemaName(asset.getSchemaName())
            .tableName(asset.getTableName())
            .tags(asset.getTags())
            .qualityScore(asset.getQualityScore())
            .columnCount(asset.getColumnCount())
            .rowCount(asset.getRowCount())
            .dataSize(asset.getDataSize())
            .createdTime(asset.getCreatedTime())
            .updatedTime(asset.getUpdatedTime())
            .searchContent(buildSearchContent(asset))
            .build();
    }

    private String buildSearchContent(Asset asset) {
        StringBuilder sb = new StringBuilder();
        sb.append(asset.getAssetName()).append(" ");
        sb.append(asset.getDescription() != null ? asset.getDescription() : "").append(" ");
        sb.append(asset.getTableName() != null ? asset.getTableName() : "").append(" ");
        sb.append(asset.getDatasourceName() != null ? asset.getDatasourceName() : "").append(" ");
        if (asset.getTags() != null) {
            sb.append(String.join(" ", asset.getTags())).append(" ");
        }
        return sb.toString().trim();
    }

    private Long getLastSyncVersion() {
        Object version = redisTemplate.opsForValue().get(SYNC_CACHE_KEY);
        return version != null ? Long.valueOf(version.toString()) : 0L;
    }

    private void updateSyncVersion() {
        redisTemplate.opsForValue().set(SYNC_CACHE_KEY, System.currentTimeMillis());
    }
}
```

```python
    # 创建索引
    index_params = {
        "index_type": "IVF_FLAT",
        "metric_type": "L2",
        "params": {"nlist": 128}
    }
    collection.create_index("embedding", index_params)
    
    # 分区
    collection.create_partition("tables")
    collection.create_partition("apis")
    collection.create_partition("files")
    
    return collection

# ============ 常用查询 ============

def search_similar_assets(collection, query_embedding, top_k=10):
    """向量相似搜索"""
    search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
    
    results = collection.search(
        data=[query_embedding],
        anns_field="embedding",
        param=search_params,
        limit=top_k,
        expr=None,
        output_fields=["asset_id", "asset_name", "description", "asset_type", "tags"]
    )
    
    return results

def search_with_filter(collection, query_embedding, asset_type, sensitivity, top_k=10):
    """带过滤条件的向量搜索"""
    search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
    
    expr = f'asset_type == "{asset_type}" && sensitivity == "{sensitivity}"'
    
    results = collection.search(
        data=[query_embedding],
        anns_field="embedding",
        param=search_params,
        limit=top_k,
        expr=expr,
        output_fields=["asset_id", "asset_name", "description", "tags"]
    )
    
    return results

def hybrid_search(collection, query_text, query_embedding, top_k=10):
    """混合搜索：向量 + 关键词"""
    # 1. 向量搜索
    vector_results = search_similar_assets(collection, query_embedding, top_k * 2)
    
    # 2. 关键词过滤
    keyword_filtered = [
        r for r in vector_results[0] 
        if query_text.lower() in r.entity.get("asset_name").lower() 
           or query_text.lower() in r.entity.get("description", "").lower()
    ]
    
    # 3. 返回TopK
    return keyword_filtered[:top_k]
```

```java
package com.enterprise.edams.knowledge.repository;

import com.enterprise.edams.knowledge.dto.VectorSearchResult;
import io.milvus.client.MilvusClient;
import io.milvus.grpc.SearchResults;
import io.milvus.param.dml.QueryParam;
import io.milvus.param.dml.SearchParam;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Repository;

import java.util.*;

/**
 * Milvus向量数据库仓储
 */
@Slf4j
@Repository
@RequiredArgsConstructor
public class VectorRepository {

    private final MilvusClient milvusClient;
    
    private static final String COLLECTION_NAME = "asset_vectors";
    private static final int TOP_K = 10;
    private static final int SEARCH_PARAM_NPROBE = 10;

    /**
     * 向量相似搜索
     */
    public List<VectorSearchResult> searchSimilarVectors(float[] queryEmbedding, int topK) {
        try {
            SearchParam searchParam = SearchParam.newBuilder()
                .withCollectionName(COLLECTION_NAME)
                .withOutFields(Arrays.asList("asset_id", "asset_name", "description", "tags"))
                .withSearchParams(Collections.singletonMap("nprobe", SEARCH_PARAM_NPROBE))
                .withTopK(topK)
                .withVectors(Collections.singletonList(queryEmbedding))
                .withVectorFieldName("embedding")
                .build();

            SearchResults results = milvusClient.search(searchParam);

            return convertSearchResults(results);
            
        } catch (Exception e) {
            log.error("向量搜索失败", e);
            return Collections.emptyList();
        }
    }

    /**
     * 带过滤条件的向量搜索
     */
    public List<VectorSearchResult> searchWithFilter(float[] queryEmbedding, 
                                                      String assetType, 
                                                      String sensitivity,
                                                      int topK) {
        try {
            String expr = String.format("asset_type == \"%s\" && sensitivity == \"%s\"", 
                assetType, sensitivity);

            SearchParam searchParam = SearchParam.newBuilder()
                .withCollectionName(COLLECTION_NAME)
                .withOutFields(Arrays.asList("asset_id", "asset_name", "description", "tags"))
                .withSearchParams(Collections.singletonMap("nprobe", SEARCH_PARAM_NPROBE))
                .withTopK(topK)
                .withVectors(Collections.singletonList(queryEmbedding))
                .withVectorFieldName("embedding")
                .withExpr(expr)
                .build();

            SearchResults results = milvusClient.search(searchParam);

            return convertSearchResults(results);
            
        } catch (Exception e) {
            log.error("带过滤条件的向量搜索失败", e);
            return Collections.emptyList();
        }
    }

    /**
     * 批量插入向量
     */
    public void insertVectors(List<Long> assetIds, List<Float> embeddings) {
        List<JSONObject> data = new ArrayList<>();
        
        for (int i = 0; i < assetIds.size(); i++) {
            JSONObject row = new JSONObject();
            row.put("id", i);
            row.put("asset_id", assetIds.get(i));
            row.put("embedding", embeddings.subList(i * 1536, (i + 1) * 1536));
            data.add(row);
        }

        try {
            InsertParam insertParam = InsertParam.newBuilder()
                .withCollectionName(COLLECTION_NAME)
                .withFields(data)
                .build();

            milvusClient.insert(insertParam);
            milvusClient.flush(COLLECTION_NAME);
            
            log.info("向量插入成功: count={}", assetIds.size());
            
        } catch (Exception e) {
            log.error("向量插入失败", e);
        }
    }

    /**
     * 删除向量
     */
    public void deleteVectors(List<Long> assetIds) {
        try {
            String expr = "asset_id in " + assetIds.stream()
                .map(String::valueOf)
                .collect(Collectors.joining(","));

            DeleteParam deleteParam = DeleteParam.newBuilder()
                .withCollectionName(COLLECTION_NAME)
                .withExpr(expr)
                .build();

            milvusClient.delete(deleteParam);
            
            log.info("向量删除成功: count={}", assetIds.size());
            
        } catch (Exception e) {
            log.error("向量删除失败", e);
        }
    }

    private List<VectorSearchResult> convertSearchResults(SearchResults results) {
        List<VectorSearchResult> searchResults = new ArrayList<>();
        
        SearchResultsWrapper wrapper = new SearchResultsWrapper(results);
        
        for int i = 0; i < wrapper.getFieldData("asset_id", 0).size(); i++ {
            VectorSearchResult result = VectorSearchResult.builder()
                .assetId(((Number) wrapper.getFieldData("asset_id", 0).get(i)).longValue())
                .assetName((String) wrapper.getFieldData("asset_name", 0).get(i))
                .description((String) wrapper.getFieldData("description", 0).get(i))
                .distance(wrapper.getDistance(0, i))
                .build();
            searchResults.add(result);
        }
        
        return searchResults;
    }
}
```
