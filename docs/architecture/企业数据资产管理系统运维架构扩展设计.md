### 2.2 异常检测引擎

```java
@Service
public class AnomalyDetectionEngine {
    
    private final TimeSeriesRepository timeSeriesRepo;
    private final MLModelRegistry modelRegistry;
    
    /**
     * 单指标异常检测
     */
    public AnomalyResult detectAnomaly(String metricName, List<DataPoint> dataPoints) {
        // 1. 数据预处理
        List<Double> values = dataPoints.stream()
            .map(DataPoint::getValue)
            .collect(Collectors.toList());
        
        // 2. 加载检测模型
        AnomalyDetectionModel model = modelRegistry.getModel(
            AnomalyModelType.ISOLATION_FOREST,
            metricName
        );
        
        // 3. 执行检测
        List<Boolean> anomalies = model.predict(values);
        
        // 4. 构建结果
        List<AnomalyPoint> anomalyPoints = new ArrayList<>();
        for (int i = 0; i < anomalies.size(); i++) {
            if (anomalies.get(i)) {
                anomalyPoints.add(AnomalyPoint.builder()
                    .timestamp(dataPoints.get(i).getTimestamp())
                    .value(dataPoints.get(i).getValue())
                    .expectedValue(model.getExpectedValue(values.get(i)))
                    .anomalyScore(model.getAnomalyScore(values.get(i)))
                    .build());
            }
        }
        
        return AnomalyResult.builder()
            .metricName(metricName)
            .totalPoints(dataPoints.size())
            .anomalyCount(anomalyPoints.size())
            .anomalyPoints(anomalyPoints)
            .detectionTime(LocalDateTime.now())
            .build();
    }
    
    /**
     * 多指标关联异常检测
     */
    public List<CorrelationAnomaly> detectCorrelationAnomaly(
            List<String> metricNames, 
            Map<String, List<DataPoint>> metricData) {
        
        // 1. 计算指标间相关性
        Map<String, Double> correlations = calculateCorrelations(metricData);
        
        // 2. 检测相关性异常
        List<CorrelationAnomaly> anomalies = new ArrayList<>();
        for (Map.Entry<String, Double> entry : correlations.entrySet()) {
            String metricPair = entry.getKey();
            Double correlation = entry.getValue();
            
            // 获取历史基线
            Double baseline = getHistoricalCorrelation(metricPair);
            
            // 检测偏差
            if (Math.abs(correlation - baseline) > 0.3) {
                anomalies.add(CorrelationAnomaly.builder()
                    .metricPair(metricPair)
                    .currentCorrelation(correlation)
                    .baselineCorrelation(baseline)
                    .deviation(correlation - baseline)
                    .severity(calculateSeverity(Math.abs(correlation - baseline)))
                    .build());
            }
        }
        
        return anomalies;
    }
}
```

```java
@Service
public class RootCauseAnalysisEngine {
    
    private final DependencyGraph dependencyGraph;
    private final AlertRepository alertRepository;
    
    /**
     * 基于依赖图的根因分析
     */
    public RCAResult analyzeRootCause(String alertId) {
        // 1. 获取告警详情
        Alert alert = alertRepository.findById(alertId);
        
        // 2. 构建告警上下文
        List<Alert> relatedAlerts = alertRepository.findRelated(
            alert.getTimeRange(),
            alert.getAffectedService()
        );
        
        // 3. 构建故障传播图
        List<DependencyEdge> propagationPath = buildPropagationPath(
            alert.getAffectedService(),
            relatedAlerts
        );
        
        // 4. 识别根因
        ServiceNode rootCause = identifyRootCause(propagationPath);
        
        // 5. 生成分析报告
        return RCAResult.builder()
            .alertId(alertId)
            .rootCause(rootCause)
            .impactScope(calculateImpactScope(rootCause))
            .propagationPath(propagationPath)
            .recommendations(generateRecommendations(rootCause))
            .confidenceScore(calculateConfidence(propagationPath))
            .analyzedAt(LocalDateTime.now())
            .build();
    }
    
    /**
     * 基于日志的根因分析
     */
    public RCAResult analyzeFromLogs(String incidentId, TimeRange timeRange) {
        // 1. 收集相关日志
        List<LogEntry> logs = logService.queryLogs(
            Query.builder()
                .timeRange(timeRange)
                .keywords(extractKeywords(incidentId))
                .level(LogLevel.ERROR)
                .build()
        );
        
        // 2. 日志聚类
        List<LogCluster> clusters = logClustering.cluster(logs);
        
        // 3. 异常日志识别
        List<LogCluster> anomalyClusters = clusters.stream()
            .filter(c -> c.getAnomalyScore() > 0.7)
            .sorted(Comparator.comparing(LogCluster::getAnomalyScore).reversed())
            .collect(Collectors.toList());
        
        // 4. 生成根因报告
        if (!anomalyClusters.isEmpty()) {
            LogCluster topCluster = anomalyClusters.get(0);
            return RCAResult.builder()
                .incidentId(incidentId)
                .rootCause(topCluster.getPattern())
                .evidence(topCluster.getSampleLogs())
                .confidenceScore(topCluster.getAnomalyScore())
                .build();
        }
        
        return RCAResult.builder()
            .incidentId(incidentId)
            .rootCause("Unable to determine root cause")
            .confidenceScore(0.0)
            .build();
    }
}
```

```java
@Service
public class SmartAlertRouting {
    
    private final AlertRepository alertRepository;
    private final NotificationService notificationService;
    
    /**
     * 智能告警路由
     */
    public void routeAlert(Alert alert) {
        // 1. 分析告警上下文
        AlertContext context = analyzeContext(alert);
        
        // 2. 确定接收人
        List<Recipient> recipients = determineRecipients(context);
        
        // 3. 选择通知渠道
        Map<Recipient, NotificationChannel> channelMap = 
            selectChannels(recipients, context);
        
        // 4. 确定优先级
        int priority = calculatePriority(context);
        
        // 5. 发送通知
        for (Recipient recipient : recipients) {
            NotificationChannel channel = channelMap.get(recipient);
            notificationService.send(Notification.builder()
                .alert(alert)
                .recipient(recipient)
                .channel(channel)
                .priority(priority)
                .build());
        }
    }
    
    /**
     * 告警上下文分析
     */
    private AlertContext analyzeContext(Alert alert) {
        // 1. 获取历史告警
        List<Alert> history = alertRepository.findRecent(
            alert.getService(),
            alert.getMetric(),
            Duration.ofDays(7)
        );
        
        // 2. 获取关联事件
        List<Event> events = eventService.findRelated(alert);
        
        // 3. 获取变更记录
        List<Change> changes = changeService.findRecent(
            alert.getService(),
            alert.getTimeRange()
        );
        
        return AlertContext.builder()
            .alert(alert)
            .history(history)
            .events(events)
            .changes(changes)
            .build();
    }
}
```

```java
@Service
public class SelfHealingEngine {
    
    private final KubernetesClient k8sClient;
    private final AlertRepository alertRepository;
    
    /**
     * 自愈执行
     */
    public HealingResult heal(HealingRequest request) {
        // 1. 验证自愈条件
        if (!canHeal(request)) {
            return HealingResult.failed("Self-healing not applicable");
        }
        
        // 2. 执行自愈动作
        HealingAction action = request.getAction();
        HealingResult result = executeAction(action, request);
        
        // 3. 验证修复效果
        if (result.isSuccess()) {
            boolean verified = verifyHealing(request);
            result.setVerified(verified);
            
            if (!verified) {
                // 回滚或升级
                result.setNextAction(upgradeAction(action));
            }
        }
        
        // 4. 记录自愈日志
        recordHealingLog(request, result);
        
        return result;
    }
    
    /**
     * 执行自愈动作
     */
    private HealingResult executeAction(HealingAction action, HealingRequest request) {
        switch (action.getType()) {
            case RESTART_POD:
                return restartPod(action.getTarget());
                
            case SCALE_UP:
                return scaleUp(action.getTarget(), action.getReplicas());
                
            case CLEANUP_DISK:
                return cleanupDisk(action.getTarget(), action.getThreshold());
                
            case RESTART_SERVICE:
                return restartService(action.getTarget());
                
            case FLUSH_CACHE:
                return flushCache(action.getTarget());
                
            default:
                return HealingResult.failed("Unknown action type");
        }
    }
    
    /**
     * Pod重启自愈
     */
    private HealingResult restartPod(String podName) {
        try {
            k8sClient.pods().inNamespace("default")
                .withName(podName)
                .delete();
            
            // 等待新Pod启动
            await().atMost(5, TimeUnit.MINUTES)
                .until(() -> isPodReady(podName));
            
            return HealingResult.success("Pod restarted successfully");
            
        } catch (Exception e) {
            return HealingResult.failed(e.getMessage());
        }
    }
}
```

### 5.1 容量预测模型

```java
@Service
public class CapacityForecastingService {
    
    /**
     * 基于时序预测的容量规划
     */
    public CapacityForecast forecast(String resourceType, 
                                   String serviceId, 
                                   Duration horizon) {
        // 1. 收集历史数据
        List<DataPoint> historicalData = collectHistoricalData(
            resourceType, serviceId, Duration.ofDays(30)
        );
        
        // 2. 选择预测模型
        ForecastingModel model = selectModel(historicalData);
        
        // 3. 执行预测
        List<ForecastPoint> forecast = model.predict(
            historicalData, horizon
        );
        
        // 4. 计算推荐容量
        CapacityRecommendation recommendation = calculateRecommendation(
            forecast, serviceId
        );
        
        return CapacityForecast.builder()
            .resourceType(resourceType)
            .serviceId(serviceId)
            .forecastPeriod(horizon)
            .forecastPoints(forecast)
            .recommendation(recommendation)
            .confidence(model.getConfidence())
            .generatedAt(LocalDateTime.now())
            .build();
    }
    
    /**
     * 计算推荐容量
     */
    private CapacityRecommendation calculateRecommendation(
            List<ForecastPoint> forecast, String serviceId) {
        
        // 1. 计算峰值
        double peakValue = forecast.stream()
            .map(ForecastPoint::getValue)
            .max(Double::compareTo)
            .orElse(0.0);
        
        // 2. 应用安全系数
        double recommendedValue = peakValue * 1.2; // 20% buffer
        
        // 3. 获取当前容量
        double currentValue = getCurrentCapacity(serviceId);
        
        // 4. 生成建议
        List<String> suggestions = new ArrayList<>();
        if (recommendedValue > currentValue * 1.3) {
            suggestions.add("建议扩容，当前容量不足");
            suggestions.add("预计需要增加 " + 
                Math.ceil((recommendedValue - currentValue) / currentValue * 100) + "%");
        } else if (recommendedValue < currentValue * 0.7) {
            suggestions.add("可以考虑缩容以节省成本");
        }
        
        return CapacityRecommendation.builder()
            .currentValue(currentValue)
            .recommendedValue(recommendedValue)
            .peakValue(peakValue)
            .suggestions(suggestions)
            .confidence(calculateRecommendationConfidence(forecast))
            .build();
    }
}
```

```java
@Service
public class ChangeManagementService {
    
    /**
     * 提交变更请求
     */
    public ChangeRequest submitChange(ChangeRequest request) {
        // 1. 变更分类
        ChangeCategory category = classifyChange(request);
        
        // 2. 风险评估
        RiskAssessment risk = assessRisk(request);
        
        // 3. 确定审批流程
        ApprovalWorkflow workflow = determineWorkflow(category, risk);
        
        // 4. 创建变更记录
        ChangeRequest savedRequest = changeRepository.save(request);
        
        // 5. 启动审批流程
        workflowService.start(workflow, savedRequest);
        
        return savedRequest;
    }
    
    /**
     * 变更风险评估
     */
    private RiskAssessment assessRisk(ChangeRequest request) {
        RiskAssessment assessment = RiskAssessment.builder()
            .changeId(request.getId())
            .build();
        
        // 评估维度
        assessment.setImpactScore(evaluateImpact(request));
        assessment.setComplexityScore(evaluateComplexity(request));
        assessment.setRollbackScore(evaluateRollback(request));
        
        // 综合风险
        double overallRisk = 
            assessment.getImpactScore() * 0.4 +
            assessment.getComplexityScore() * 0.3 +
            assessment.getRollbackScore() * 0.3;
        
        assessment.setOverallRisk(overallRisk);
        assessment.setRiskLevel(determineRiskLevel(overallRisk));
        
        return assessment;
    }
}
```
