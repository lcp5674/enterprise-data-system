## 1. 代码规范

### 1.1 命名规范

#### 1.1.1 通用命名原则

| 原则 | 说明 |
|------|------|
| **清晰性** | 名称应清晰表达意图，避免缩写和歧义 |
| **一致性** | 相同概念在不同地方使用相同命名 |
| **简洁性** | 在保证清晰的前提下保持简洁 |
| **可搜索性** | 避免使用单个字符或通用词作为名称 |

### 1.3 代码格式规范

#### 1.3.1 类结构顺序

```java
package com.enterprise.edams.asset;

import lombok.extern.slf4j.Slf4j;

/**
 * 数据资产管理服务实现
 * 
 * 功能描述：负责数据资产的CRUD、血缘管理、质量检测等核心业务逻辑
 * 
 * @author Architecture Team
 * @version 1.0.0
 */
@Slf4j
@RestController
@RequestMapping("/api/v1/assets")
@RequiredArgsConstructor
@Tag(name = "资产管理", description = "数据资产管理相关接口")
public class DataAssetController {

    // ============ 常量 ============
    private static final int MAX_BATCH_SIZE = 1000;
    private static final String CACHE_PREFIX = "asset:";
    
    // ============ 依赖注入 ============
    private final DataAssetService dataAssetService;
    private final LineageService lineageService;
    private final QualityService qualityService;
    private final AuditService auditService;
    
    // ============ 业务方法 ============
    
    @PostMapping
    @Operation(summary = "创建数据资产", description = "创建新的数据资产记录")
    public Result<DataAssetVO> create(@Valid @RequestBody CreateAssetRequest request) {
        // 1. 记录审计日志
        auditService.logAction(AuditAction.CREATE, "data_asset", null);
        
        // 2. 执行创建
        DataAsset asset = dataAssetService.create(request);
        
        // 3. 构建返回
        return Result.success(DataAssetConvert.toVO(asset));
    }
    
    @GetMapping("/{id}")
    @Operation(summary = "获取资产详情", description = "根据ID获取数据资产详细信息")
    public Result<DataAssetDetailVO> getById(@PathVariable Long id) {
        DataAsset asset = dataAssetService.getById(id);
        return Result.success(DataAssetConvert.toDetailVO(asset));
    }
    
    // ============ 私有方法 ============
    
    private void validateCreateRequest(CreateAssetRequest request) {
        if (request == null) {
            throw new IllegalArgumentException("请求参数不能为空");
        }
        if (StringUtils.isBlank(request.getAssetName())) {
            throw new IllegalArgumentException("资产名称不能为空");
        }
    }
    
    private DataAssetVO convertToVO(DataAsset asset) {
        return DataAssetConvert.toVO(asset);
    }
}
```

## 2. Git分支管理策略

### 2.1 分支模型选择

**采用 Trunk-Based Development 为主 + GitFlow 为辅的混合策略**

```
适用场景：
- 主开发分支（main）：Trunk-Based
- 特性开发、功能发布：GitFlow
```

### 2.2 分支命名规范

```bash
# ============ 分支类型 ============

# 特性分支
feature/{功能域}-{简短描述}
feature/asset-data-lineage              # 资产血缘功能
feature/quality-auto-check              # 质量自动检测
feature/security-data-masking           # 数据脱敏功能

# 缺陷修复分支
bugfix/{模块}-{问题描述}
bugfix/auth-login-timeout                # 登录超时问题
bugfix/asset-search-null                 # 资产搜索空指针

# 发布分支
release/v{版本号}
release/v1.0.0
release/v1.1.0

# 热修复分支
hotfix/{问题描述}
hotfix/critical-security-patch            # 关键安全修复

# 技术债务/重构
refactor/{模块}-{描述}
refactor/dao-layer-optimization           # DAO层优化
techdebt/remove-deprecated-api           # 移除废弃API

# 文档更新
docs/{模块}-{描述}
docs/api-authentication                  # 认证API文档
docs/readme-update                       # README更新
```

### 2.3 分支操作流程

```bash
# ============ 开发流程 ============

# 1. 从main创建特性分支
git checkout main
git pull origin main
git checkout -b feature/asset-share

# 2. 开发与提交
git add .
git commit -m "feat(asset): 实现资产分享功能

- 添加分享链接生成逻辑
- 添加分享权限验证
- 添加分享记录存储
- 关联issue: #EDAMS-123"

# 3. 定期同步main最新代码
git fetch origin
git rebase origin/main

# 4. 完成后发起PR/MR
git push origin feature/asset-share

# 5. Code Review通过后合并到main
git checkout main
git pull origin main
git merge feature/asset-share
git push origin main

# 6. 删除已合并的分支
git branch -d feature/asset-share
git push origin --delete feature/asset-share
```

## 3. 提交信息规范

### 3.1 Commit Message格式

**采用 Conventional Commits 规范**

```
<type>(<scope>): <subject>

<body>

<footer>
```

### 3.2 Type类型定义

```bash
# ============ Type 类型 ============

feat:     # 新功能 - 表示新增了一个功能
fix:      # 缺陷修复 - 表示修复了一个bug
docs:     # 文档变更 - 仅文档变更
style:    # 格式调整 - 不影响代码含义的变更（空格、格式化、分号等）
refactor: # 重构 - 不修复bug也不新增功能的代码变更
perf:     # 性能优化 - 提升性能的代码变更
test:     # 测试 - 添加或修改测试代码
build:    # 构建 - 构建系统或依赖变更
ci:       # CI/CD - CI配置文件和脚本变更
chore:    # 杂项 - 不修改src或test的其他变更
revert:   # 回退 - 回退到上一个版本
```

```java
    private T data;          // 响应数据
    private long timestamp;  // 时间戳

            .message("操作成功")
            .data(data)
            .timestamp(System.currentTimeMillis())
            .build();
    }

        return Result.<T>builder()
            .code(500)
            .message(message)
            .timestamp(System.currentTimeMillis())
            .build();
    }

        return Result.<T>builder()
            .code(code)
            .message(message)
            .timestamp(System.currentTimeMillis())
            .build();
    }
}
```

### 6.3 SQL编写规范

```sql
-- ============ SQL编写规范 ============

-- 1. 字段列表明确，避免使用*
-- ❌ 不推荐
SELECT * FROM data_asset WHERE owner_id = 1;

-- ✅ 推荐
SELECT id, asset_name, asset_type, status, created_time 
FROM data_asset 
WHERE owner_id = 1;

-- 2. WHERE条件明确
-- ❌ 不推荐
SELECT * FROM data_asset WHERE deleted = 0;

-- ✅ 推荐（显式条件）
SELECT id, asset_name 
FROM data_asset 
WHERE deleted = 0 
  AND status = 'PUBLISHED'
  AND sensitivity != 'SECRET';

-- 3. 使用LIMIT限制
-- ❌ 避免大结果集
SELECT * FROM data_asset;

-- ✅ 推荐
SELECT * FROM data_asset LIMIT 100;

-- 4. 避免SELECT N+1
-- ❌ 不推荐（多次查询）
SELECT * FROM data_asset WHERE owner_id = 1;
SELECT * FROM user WHERE id IN (1, 2, 3);
SELECT * FROM department WHERE id IN (10, 20);

-- ✅ 推荐（JOIN或IN子句一次性获取）
SELECT a.*, u.user_name, d.dept_name
FROM data_asset a
LEFT JOIN user u ON a.owner_id = u.id
LEFT JOIN department d ON u.dept_id = d.id
WHERE a.owner_id = 1;

-- 5. 分页查询优化
-- ❌ 不推荐（深度分页）
SELECT * FROM data_asset 
ORDER BY created_time DESC 
LIMIT 1000000, 20;

-- ✅ 推荐（基于游标的分页）
SELECT * FROM data_asset 
WHERE created_time < '2024-01-01'
  AND id < #{lastId}
ORDER BY created_time DESC, id DESC
LIMIT 20;

-- 6. 批量操作
-- ❌ 避免循环单条插入
INSERT INTO asset_tag (asset_id, tag_id) VALUES (1, 1);
INSERT INTO asset_tag (asset_id, tag_id) VALUES (1, 2);

-- ✅ 推荐
INSERT INTO asset_tag (asset_id, tag_id) VALUES 
(1, 1), (1, 2), (1, 3), (1, 4);

-- 7. 敏感数据脱敏
-- ❌ 避免明文存储和传输
SELECT email, phone FROM user;

-- ✅ 推荐（脱敏处理）
SELECT 
    id,
    name,
    CONCAT(LEFT(email, 2), '***', SUBSTRING_INDEX(email, '@', -1)) AS email,
    CONCAT(LEFT(phone, 3), '****', RIGHT(phone, 4)) AS phone
FROM user;
```

### 8.2 统一异常处理

```java
@RestControllerAdvice
@Slf4j
public class GlobalExceptionHandler {

    // ============ 业务异常 ============
    @ExceptionHandler(BusinessException.class)
    public Result<Void> handleBusinessException(BusinessException e) {
        log.warn("业务异常: code={}, msg={}", e.getCode(), e.getMessage());
        return Result.fail(e.getCode(), e.getMessage());
    }
    
    // ============ 参数校验异常 ============
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public Result<Void> handleValidationException(MethodArgumentNotValidException e) {
        String message = e.getBindingResult().getFieldErrors().stream()
            .map(FieldError::getDefaultMessage)
            .collect(Collectors.joining("; "));
        
        log.warn("参数校验失败: {}", message);
        return Result.fail(400, message);
    }
    
    // ============ 资源不存在 ============
    @ExceptionHandler(NoSuchElementException.class)
    public Result<Void> handleNoSuchElementException(NoSuchElementException e) {
        log.warn("资源不存在: {}", e.getMessage());
        return Result.fail(404, "请求的资源不存在");
    }
    
    // ============ 权限异常 ============
    @ExceptionHandler(AccessDeniedException.class)
    public Result<Void> handleAccessDeniedException(AccessDeniedException e) {
        log.warn("访问被拒绝: {}", e.getMessage());
        return Result.fail(403, "没有访问权限");
    }
    
    // ============ 系统异常 ============
    @ExceptionHandler(Exception.class)
    public Result<Void> handleException(Exception e) {
        log.error("系统异常", e);
        return Result.fail(500, "系统内部错误，请稍后重试");
    }
}
```

### 8.3 异常处理原则

```java
public class DataAssetService {

    // ============ 异常处理原则 ============
    
    // 1. 不要捕获所有异常后只打印日志
    public void process(Long id) {
        try {
            // 处理逻辑
        } catch (Exception e) {
            // ❌ 避免：捕获所有异常
            log.error("处理失败", e);
        }
    }
    
    // ✅ 正确做法
    public void process(Long id) {
        try {
            // 处理逻辑
        } catch (SpecificException e) {
            // 处理特定异常
            throw new BusinessException(CODE, e.getMessage());
        } catch (UnexpectedException e) {
            // 未知异常记录后重新抛出
            log.error("Unexpected error", e);
            throw new SystemException("处理失败", e);
        }
    }
    
    // 2. 不要吞掉异常
    public void process1(Long id) {
        try {
            // 处理逻辑
        } catch (Exception e) {
            // ❌ 避免：吞掉异常
            log.info("出错了", e);
        }
    }
    
    // ✅ 正确：要么处理，要么重新抛出
    public void process2(Long id) {
        try {
            // 处理逻辑
        } catch (Exception e) {
            log.error("处理失败", e);
            throw new BusinessException(CODE, "处理失败");
        }
    }
    
    // 3. 异常链传递
    public void service1() {
        try {
            dao.query();
        } catch (SQLException e) {
            // ✅ 保留原始异常信息
            throw new SystemException("查询失败", e);
        }
    }
    
    // 4. 使用Optional避免空指针
    public String getAssetName(Long id) {
        // ❌ 避免
        return repository.findById(id).get().getAssetName();
        
        // ✅ 推荐
        return repository.findById(id)
            .map(DataAsset::getAssetName)
            .orElseThrow(() -> new AssetNotFoundException(id));
    }
}
```

```java
public class DataAssetRepository {

    public void validateType(String type) {
        if (!ALLOWED_TYPES.contains(type)) {
            throw new BusinessException("无效的资产类型");
        }
    }
```
