### 1.3 API调用规格

#### 1.3.1 DeepSeek API调用

```java
package com.enterprise.edams.ai.service.impl;

import com.enterprise.edams.ai.config.ModelConfig;
import com.enterprise.edams.ai.dto.*;
import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.*;
import org.springframework.stereotype.Service;
import org.springframework.web.reactive.function.client.WebClient;
import reactor.core.publisher.Flux;
import reactor.core.publisher.Mono;

import java.time.Duration;
import java.util.*;

/**
 * DeepSeek模型实现
 */
@Slf4j
@Service
public class DeepSeekServiceImpl implements LLMService {

    private final WebClient webClient;
    private final ObjectMapper objectMapper;
    private final ModelConfig modelConfig;

    public DeepSeekServiceImpl(
            @Value("${ai.models.deepseek-chat.api-url}") String apiUrl,
            @Value("${ai.models.deepseek-chat.api-key}") String apiKey,
            ObjectMapper objectMapper,
            ModelConfig modelConfig) {
        
        this.webClient = WebClient.builder()
            .baseUrl(apiUrl)
            .defaultHeader(HttpHeaders.AUTHORIZATION, "Bearer " + apiKey)
            .defaultHeader(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_JSON_VALUE)
            .build();
        this.objectMapper = objectMapper;
        this.modelConfig = modelConfig;
    }

    @Override
    public LLMResponse generate(LLMRequest request) {
        log.info("DeepSeek文本生成: model={}, promptLength={}", 
            request.getModel(), request.getPrompt().length());

        long startTime = System.currentTimeMillis();

        try {
            // 构建请求体
            Map<String, Object> requestBody = buildRequestBody(request);

            // 调用API
            String response = webClient.post()
                .uri("/chat/completions")
                .bodyValue(requestBody)
                .retrieve()
                .bodyToMono(String.class)
                .timeout(Duration.ofMillis(60000))
                .block();

            // 解析响应
            LLMResponse llmResponse = parseResponse(response);
            
            long duration = System.currentTimeMillis() - startTime;
            log.info("DeepSeek调用成功: duration={}ms, tokens={}", 
                duration, llmResponse.getUsage().getTotalTokens());

            // 记录成本
            recordCost(request.getModel(), llmResponse.getUsage().getTotalTokens());

            return llmResponse;

        } catch (Exception e) {
            log.error("DeepSeek调用失败", e);
            throw new AIException("LLM调用失败: " + e.getMessage(), e);
        }
    }

    @Override
    public Flux<String> generateStream(LLMRequest request) {
        log.info("DeepSeek流式生成: model={}", request.getModel());

        Map<String, Object> requestBody = buildRequestBody(request);
        requestBody.put("stream", true);

        return webClient.post()
            .uri("/chat/completions")
            .bodyValue(requestBody)
            .retrieve()
            .bodyToMono(String.class)
            .timeout(Duration.ofMillis(300000))
            .flatMapMany(response -> Flux.fromArray(response.split("\n")))
            .filter(line -> line.startsWith("data: "))
            .map(line -> line.substring(6))
            .filter(line -> !"[DONE]".equals(line))
            .map(this::extractDeltaContent);
    }

    @Override
    public List<LLMResponse> batchGenerate(List<LLMRequest> requests) {
        return requests.stream()
            .map(this::generate)
            .toList();
    }

    @Override
    public EmbeddingResponse embed(EmbeddingRequest request) {
        log.info("DeepSeek嵌入生成: model={}, textLength={}", 
            request.getModel(), request.getText().length());

        long startTime = System.currentTimeMillis();

        Map<String, Object> requestBody = Map.of(
            "model", request.getModel() != null ? request.getModel() : "text-embedding-ada-002",
            "input", request.getText()
        );

        try {
            String response = webClient.post()
                .uri("/embeddings")
                .bodyValue(requestBody)
                .retrieve()
                .bodyToMono(String.class)
                .timeout(Duration.ofMillis(30000))
                .block();

            EmbeddingResponse embeddingResponse = parseEmbeddingResponse(response);

            long duration = System.currentTimeMillis() - startTime;
            log.info("嵌入生成成功: duration={}ms", duration);

            return embeddingResponse;

        } catch (Exception e) {
            log.error("嵌入生成失败", e);
            throw new AIException("嵌入生成失败: " + e.getMessage(), e);
        }
    }

    private Map<String, Object> buildRequestBody(LLMRequest request) {
        Map<String, Object> body = new HashMap<>();
        
        body.put("model", request.getModel() != null ? request.getModel() : "deepseek-chat");
        
        // 构建消息
        List<Map<String, String>> messages = new ArrayList<>();
        
        // 系统提示词
        if (request.getSystemPrompt() != null) {
            messages.add(Map.of(
                "role", "system",
                "content", request.getSystemPrompt()
            ));
        }
        
        // 用户提示词
        messages.add(Map.of(
            "role", "user",
            "content", request.getPrompt()
        ));
        
        body.put("messages", messages);
        
        // 生成参数
        body.put("max_tokens", request.getMaxTokens() != null ? request.getMaxTokens() : 8192);
        body.put("temperature", request.getTemperature() != null ? request.getTemperature() : 0.7);
        body.put("top_p", request.getTopP() != null ? request.getTopP() : 0.95);
        body.put("stream", false);
        
        return body;
    }

    private LLMResponse parseResponse(String responseBody) {
        try {
            JsonNode root = objectMapper.readTree(responseBody);
            
            String content = root.path("choices")
                .get(0)
                .path("message")
                .path("content")
                .asText();
            
            JsonNode usage = root.path("usage");
            
            UsageInfo usageInfo = UsageInfo.builder()
                .promptTokens(usage.path("prompt_tokens").asInt())
                .completionTokens(usage.path("completion_tokens").asInt())
                .totalTokens(usage.path("total_tokens").asInt())
                .build();
            
            return LLMResponse.builder()
                .content(content)
                .model(root.path("model").asText())
                .usage(usageInfo)
                .finishReason(root.path("choices").get(0).path("finish_reason").asText())
                .build();
                
        } catch (Exception e) {
            log.error("响应解析失败: {}", responseBody, e);
            throw new AIException("响应解析失败", e);
        }
    }

    private EmbeddingResponse parseEmbeddingResponse(String responseBody) {
        try {
            JsonNode root = objectMapper.readTree(responseBody);
            
            List<Double> embedding = new ArrayList<>();
            root.path("data").get(0).path("embedding")
                .forEach(node -> embedding.add(node.asDouble()));
            
            return EmbeddingResponse.builder()
                .embedding(embedding)
                .model(root.path("model").asText())
                .tokens(root.path("usage").path("total_tokens").asInt())
                .build();
                
        } catch (Exception e) {
            log.error("嵌入响应解析失败: {}", responseBody, e);
            throw new AIException("嵌入响应解析失败", e);
        }
    }

    private String extractDeltaContent(String sseData) {
        try {
            if (sseData == null || sseData.isEmpty()) {
                return "";
            }
            JsonNode node = objectMapper.readTree(sseData);
            return node.path("choices")
                .get(0)
                .path("delta")
                .path("content")
                .asText("");
        } catch (Exception e) {
            return "";
        }
    }

    private void recordCost(String model, int tokens) {
        // 成本记录逻辑
        // DeepSeek价格: ¥0.001/千tokens
        double cost = tokens * 0.000001;  // 每token价格
        log.debug("LLM成本记录: model={}, tokens={}, cost={}", model, tokens, cost);
    }
}
```

### 1.5 降级策略

```java
package com.enterprise.edams.ai.service;

import com.enterprise.edams.ai.dto.*;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import reactor.core.publisher.Flux;

import java.util.List;

/**
 * LLM服务降级处理
 */
@Slf4j
@Service
public class FallbackLLMService implements LLMService {

    private static final String FALLBACK_RESPONSE = 
        "抱歉，当前服务繁忙，请稍后再试。您可以尝试以下操作：\n" +
        "1. 简化您的问题\n" +
        "2. 稍后重试\n" +
        "3. 联系管理员检查服务状态";

    @Override
    public LLMResponse generate(LLMRequest request) {
        log.warn("LLM服务降级，返回默认响应");
        
        return LLMResponse.builder()
            .content(FALLBACK_RESPONSE)
            .model("fallback")
            .usage(UsageInfo.builder()
                .promptTokens(0)
                .completionTokens(FALLBACK_RESPONSE.length() / 4)
                .totalTokens(FALLBACK_RESPONSE.length() / 4)
                .build())
            .finishReason("fallback")
            .build();
    }

    @Override
    public Flux<String> generateStream(LLMRequest request) {
        return Flux.just(FALLBACK_RESPONSE);
    }

    @Override
    public List<LLMResponse> batchGenerate(List<LLMRequest> requests) {
        return requests.stream()
            .map(r -> generate(r))
            .toList();
    }

    @Override
    public EmbeddingResponse embed(EmbeddingRequest request) {
        // 返回零向量作为降级
        List<Double> zeroEmbedding = java.util.Collections.nCopies(1536, 0.0);
        
        return EmbeddingResponse.builder()
            .embedding(zeroEmbedding)
            .model("fallback")
            .tokens(0)
            .build();
    }
}
```

### 2.2 Java集成代码

```java
package com.enterprise.edams.storage.service;

import com.enterprise.edams.storage.config.MinioConfig;
import com.enterprise.edams.storage.dto.*;
import io.minio.*;
import io.minio.http.Method;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;

import java.io.*;
import java.nio.file.Files;
import java.time.Duration;
import java.util.*;

/**
 * MinIO对象存储服务
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class MinioStorageService {

    private final MinioClient minioClient;
    private final MinioConfig minioConfig;

    // ============ Bucket操作 ============

    /**
     * 创建Bucket
     */
    public void createBucket(String bucketName) {
        try {
            boolean exists = minioClient.bucketExists(
                BucketExistsArgs.builder().bucket(bucketName).build()
            );
            
            if (!exists) {
                minioClient.makeBucket(
                    MakeBucketArgs.builder().bucket(bucketName).build()
                );
                log.info("Bucket创建成功: {}", bucketName);
                
                // 设置存储策略
                setBucketPolicy(bucketName);
            }
        } catch (Exception e) {
            log.error("Bucket创建失败: {}", bucketName, e);
            throw new StorageException("Bucket创建失败: " + e.getMessage(), e);
        }
    }

    /**
     * 设置Bucket策略
     */
    private void setBucketPolicy(String bucketName) {
        try {
            String policy = """
                {
                    "Version": "2012-10-17",
                    "Statement": [
                        {
                            "Effect": "Allow",
                            "Principal": {"AWS": "*"},
                            "Action": ["s3:GetObject"],
                            "Resource": "arn:aws:s3:::%s/*"
                        }
                    ]
                }
                """.formatted(bucketName);
            
            minioClient.setBucketPolicy(
                SetBucketPolicyArgs.builder()
                    .bucket(bucketName)
                    .config(policy)
                    .build()
            );
        } catch (Exception e) {
            log.warn("设置Bucket策略失败", e);
        }
    }

    // ============ 文件上传 ============

    /**
     * 上传文件
     */
    public StorageResult uploadFile(UploadRequest request) {
        String bucketName = request.getBucket() != null ? 
            request.getBucket() : minioConfig.getBucket().getAssets();
        
        // 确保Bucket存在
        createBucket(bucketName);
        
        String objectName = generateObjectName(request);
        
        try {
            InputStream inputStream;
            String contentType;
            long fileSize;
            
            if (request.getFile() != null) {
                MultipartFile file = request.getFile();
                inputStream = file.getInputStream();
                contentType = file.getContentType();
                fileSize = file.getSize();
            } else if (request.getData() != null) {
                inputStream = new ByteArrayInputStream(request.getData());
                contentType = request.getContentType() != null ? 
                    request.getContentType() : "application/octet-stream";
                fileSize = request.getData().length;
            } else {
                throw new IllegalArgumentException("文件数据不能为空");
            }
            
            // 上传
            minioClient.putObject(
                PutObjectArgs.builder()
                    .bucket(bucketName)
                    .object(objectName)
                    .stream(inputStream, fileSize, -1)
                    .contentType(contentType)
                    .headers(buildHeaders(request))
                    .build()
            );
            
            log.info("文件上传成功: bucket={}, object={}, size={}", 
                bucketName, objectName, fileSize);
            
            // 获取文件信息
            StatObjectResponse stat = minioClient.statObject(
                StatObjectArgs.builder()
                    .bucket(bucketName)
                    .object(objectName)
                    .build()
            );
            
            return StorageResult.builder()
                .bucket(bucketName)
                .objectName(objectName)
                .etag(stat.etag())
                .size(stat.size())
                .contentType(contentType)
                .url(getPresignedUrl(bucketName, objectName))
                .uploadTime(new Date())
                .build();
                
        } catch (Exception e) {
            log.error("文件上传失败: bucket={}, object={}", bucketName, objectName, e);
            throw new StorageException("文件上传失败: " + e.getMessage(), e);
        }
    }

    /**
     * 分片上传
     */
    public UploadIdResult initiateMultipartUpload(String bucketName, String objectName) {
        try {
            String uploadId = minioClient.initMultipartUpload(
                InitMultipartUploadArgs.builder()
                    .bucket(bucketName)
                    .object(objectName)
                    .build()
            ).uploadId();
            
            log.info("分片上传初始化成功: bucket={}, object={}, uploadId={}", 
                bucketName, objectName, uploadId);
            
            return UploadIdResult.builder()
                .bucket(bucketName)
                .objectName(objectName)
                .uploadId(uploadId)
                .build();
                
        } catch (Exception e) {
            log.error("分片上传初始化失败", e);
            throw new StorageException("分片上传初始化失败", e);
        }
    }

    /**
     * 上传分片
     */
    public PartETag uploadPart(UploadPartRequest request) {
        try {
            ObjectWriteResponse response = minioClient.putObject(
                PutObjectArgs.builder()
                    .bucket(request.getBucket())
                    .object(request.getObjectName())
                    .stream(request.getInputStream(), request.getSize(), -1)
                    .uploadId(request.getUploadId())
                    .partNumber(request.getPartNumber())
                    .build()
            );
            
            return PartETag.builder()
                .partNumber(request.getPartNumber())
                .etag(response.etag())
                .build();
                
        } catch (Exception e) {
            log.error("分片上传失败: partNumber={}", request.getPartNumber(), e);
            throw new StorageException("分片上传失败", e);
        }
    }

    /**
     * 完成分片上传
     */
    public CompleteMultipartUploadResult completeMultipartUpload(
            String bucketName, String objectName, String uploadId, List<PartETag> parts) {
        try {
            ObjectWriteResponse response = minioClient.completeMultipartUpload(
                CompleteMultipartUploadArgs.builder()
                    .bucket(bucketName)
                    .object(objectName)
                    .uploadId(uploadId)
                    .parts(parts.stream()
                        .map(p -> io.minio.ComposeSource.builder()
                            .bucket(bucketName)
                            .object(objectName)
                            .partNumber(p.getPartNumber())
                            .etag(p.getEtag())
                            .build())
                        .toList())
                    .build()
            );
            
            log.info("分片上传完成: bucket={}, object={}", bucketName, objectName);
            
            return CompleteMultipartUploadResult.builder()
                .bucket(bucketName)
                .objectName(objectName)
                .etag(response.etag())
                .build();
                
        } catch (Exception e) {
            log.error("分片上传完成失败", e);
            throw new StorageException("分片上传完成失败", e);
        }
    }

    // ============ 文件下载 ============

    /**
     * 下载文件
     */
    public InputStream downloadFile(String bucketName, String objectName) {
        try {
            return minioClient.getObject(
                GetObjectArgs.builder()
                    .bucket(bucketName)
                    .object(objectName)
                    .build()
            );
        } catch (Exception e) {
            log.error("文件下载失败: bucket={}, object={}", bucketName, objectName, e);
            throw new StorageException("文件下载失败", e);
        }
    }

    /**
     * 下载文件到本地
     */
    public void downloadToFile(String bucketName, String objectName, String localPath) {
        try {
            minioClient.downloadObject(
                DownloadObjectArgs.builder()
                    .bucket(bucketName)
                    .object(objectName)
                    .filename(localPath)
                    .build()
            );
            log.info("文件下载到本地: bucket={}, object={}, path={}", 
                bucketName, objectName, localPath);
        } catch (Exception e) {
            log.error("文件下载失败", e);
            throw new StorageException("文件下载失败", e);
        }
    }

    // ============ 文件管理 ============

    /**
     * 获取文件信息
     */
    public StorageInfo getFileInfo(String bucketName, String objectName) {
        try {
            StatObjectResponse stat = minioClient.statObject(
                StatObjectArgs.builder()
                    .bucket(bucketName)
                    .object(objectName)
                    .build()
            );
            
            return StorageInfo.builder()
                .bucket(bucketName)
                .objectName(objectName)
                .etag(stat.etag())
                .size(stat.size())
                .contentType(stat.contentType())
                .lastModified(stat.lastModified())
                .build();
                
        } catch (Exception e) {
            log.error("获取文件信息失败", e);
            throw new StorageException("获取文件信息失败", e);
        }
    }

    /**
     * 删除文件
     */
    public void deleteFile(String bucketName, String objectName) {
        try {
            minioClient.removeObject(
                RemoveObjectArgs.builder()
                    .bucket(bucketName)
                    .object(objectName)
                    .build()
            );
            log.info("文件删除成功: bucket={}, object={}", bucketName, objectName);
        } catch (Exception e) {
            log.error("文件删除失败", e);
            throw new StorageException("文件删除失败", e);
        }
    }

    /**
     * 批量删除
     */
    public void deleteFiles(String bucketName, List<String> objectNames) {
        try {
            List<DeleteObject> objects = objectNames.stream()
                .map(name -> new DeleteObject(name))
                .toList();
            
            minioClient.removeObjects(
                RemoveObjectsArgs.builder()
                    .bucket(bucketName)
                    .objects(objects)
                    .build()
            );
            log.info("批量删除成功: bucket={}, count={}", bucketName, objectNames.size());
        } catch (Exception e) {
            log.error("批量删除失败", e);
            throw new StorageException("批量删除失败", e);
        }
    }

    /**
     * 复制文件
     */
    public void copyFile(String sourceBucket, String sourceObject, 
                        String destBucket, String destObject) {
        try {
            minioClient.copyObject(
                CopyObjectArgs.builder()
                    .source(CopySource.builder()
                        .bucket(sourceBucket)
                        .object(sourceObject)
                        .build())
                    .bucket(destBucket)
                    .object(destObject)
                    .build()
            );
            log.info("文件复制成功: {} -> {}", sourceObject, destObject);
        } catch (Exception e) {
            log.error("文件复制失败", e);
            throw new StorageException("文件复制失败", e);
        }
    }

    // ============ 访问URL ============

    /**
     * 获取预签名URL
     */
    public String getPresignedUrl(String bucketName, String objectName) {
        return getPresignedUrl(bucketName, objectName, Duration.ofHours(1));
    }

    /**
     * 获取带过期时间的预签名URL
     */
    public String getPresignedUrl(String bucketName, String objectName, Duration expiry) {
        try {
            return minioClient.getPresignedObjectUrl(
                GetPresignedObjectUrlArgs.builder()
                    .method(Method.GET)
                    .bucket(bucketName)
                    .object(objectName)
                    .expiry((int) expiry.toSeconds())
                    .build()
            );
        } catch (Exception e) {
            log.error("获取预签名URL失败", e);
            throw new StorageException("获取预签名URL失败", e);
        }
    }

    // ============ 生命周期 ============

    /**
     * 设置对象过期时间
     */
    public void setObjectExpiry(String bucketName, String objectName, Duration expiry) {
        try {
            minioClient.setObjectRetention(
                SetObjectRetentionArgs.builder()
                    .bucket(bucketName)
                    .object(objectName)
                    .bypassGovernanceMode(true)
                    .build()
            );
        } catch (Exception e) {
            log.warn("设置对象过期失败", e);
        }
    }

    // ============ 工具方法 ============

    private String generateObjectName(UploadRequest request) {
        String prefix = request.getPrefix() != null ? 
            request.getPrefix() + "/" : "";
        String fileName = request.getFile() != null ? 
            request.getFile().getOriginalFilename() : 
            UUID.randomUUID().toString();
        String timestamp = String.valueOf(System.currentTimeMillis());
        return prefix + timestamp + "_" + fileName;
    }

    private Map<String, String> buildHeaders(UploadRequest request) {
        Map<String, String> headers = new HashMap<>();
        if (request.getUserMetadata() != null) {
            request.getUserMetadata().forEach(headers::put);
        }
        return headers;
    }
}
```

```java
package com.enterprise.edams.mq.registry;

import com.enterprise.edams.mq.config.SchemaRegistryConfig;
import io.confluent.kafka.schemaregistry.client.*;
import io.confluent.kafka.schemaregistry.client.rest.exceptions.*;
import io.confluent.kafka.schemaregistry.avro.*;
import lombok.extern.slf4j.Slf4j;
import org.apache.avro.Schema;
import org.springframework.stereotype.Service;

import java.util.*;

/**
 * Schema Registry服务
 */
@Slf4j
@Service
public class SchemaRegistryService {

    private final SchemaRegistryClient schemaRegistryClient;
    private final SchemaRegistryConfig config;

    public SchemaRegistryService(SchemaRegistryConfig config) {
        this.config = config;
        this.schemaRegistryClient = new CachedSchemaRegistryClient(
            config.getUrl(), 
            config.getMaxSchemasPerSubject()
        );
    }

    // ============ Schema注册 ============

    /**
     * 注册Avro Schema
     */
    public int registerSchema(String subject, Schema schema) {
        try {
            ParsedSchema parsedSchema = new AvroSchema(schema);
            int schemaId = schemaRegistryClient.register(subject, parsedSchema);
            log.info("Schema注册成功: subject={}, id={}", subject, schemaId);
            return schemaId;
        } catch (RestClientException e) {
            log.error("Schema注册失败: subject={}", subject, e);
            throw new SchemaRegistryException("Schema注册失败", e);
        }
    }

    /**
     * 注册JSON Schema
     */
    public int registerJsonSchema(String subject, String schemaJson) {
        try {
            ParsedSchema parsedSchema = new JsonSchema(schemaJson);
            int schemaId = schemaRegistryClient.register(subject, parsedSchema);
            log.info("JSON Schema注册成功: subject={}, id={}", subject, schemaId);
            return schemaId;
        } catch (Exception e) {
            log.error("JSON Schema注册失败", e);
            throw new SchemaRegistryException("JSON Schema注册失败", e);
        }
    }

    // ============ Schema查询 ============

    /**
     * 获取最新Schema
     */
    public RegisteredSchema getLatestSchema(String subject) {
        try {
            return schemaRegistryClient.getLatestSchema(subject);
        } catch (RestClientException e) {
            if (e.getStatusCode() == 404) {
                return null;
            }
            throw new SchemaRegistryException("获取Schema失败", e);
        }
    }

    /**
     * 根据ID获取Schema
     */
    public String getSchemaById(int schemaId) {
        try {
            return schemaRegistryClient.getById(schemaId).rawSchema();
        } catch (RestClientException e) {
            throw new SchemaRegistryException("获取Schema失败", e);
        }
    }

    /**
     * 获取所有版本
     */
    public List<Integer> getAllVersions(String subject) {
        try {
            return schemaRegistryClient.getAllVersions(subject);
        } catch (RestClientException e) {
            throw new SchemaRegistryException("获取版本列表失败", e);
        }
    }

    // ============ Schema兼容检查 ============

    /**
     * 检查兼容性
     */
    public boolean checkCompatibility(String subject, Schema newSchema) {
        try {
            return schemaRegistryClient.testCompatibility(
                subject, 
                new AvroSchema(newSchema)
            );
        } catch (RestClientException e) {
            log.warn("兼容性检查失败: {}", e.getMessage());
            return false;
        }
    }

    /**
     * 更新兼容性策略
     */
    public void updateCompatibility(String subject, String compatibility) {
        try {
            schemaRegistryClient.updateCompatibility(subject, compatibility);
            log.info("兼容性策略更新: subject={}, compatibility={}", subject, compatibility);
        } catch (RestClientException e) {
            throw new SchemaRegistryException("更新兼容性策略失败", e);
        }
    }

    // ============ 主题Schema配置 ============

    /**
     * 为主题配置Schema
     */
    public void configureTopicSchema(String topic, Schema schema, String compatibility) {
        String subject = topic + "-value";
        
        // 检查是否已注册
        RegisteredSchema existing = getLatestSchema(subject);
        
        if (existing == null) {
            // 首次注册
            registerSchema(subject, schema);
            updateCompatibility(subject, compatibility);
        } else {
            // 检查兼容性
            if (!checkCompatibility(subject, schema)) {
                throw new SchemaIncompatibleException(
                    "新Schema与现有Schema不兼容"
                );
            }
            // 注册新版本
            registerSchema(subject, schema);
        }
    }
}
```

```java
package com.enterprise.edams.datasource.service;

import com.enterprise.edams.datasource.config.DatasourceConfig;
import com.enterprise.edams.datasource.dto.ConnectionTestResult;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.sql.Connection;
import java.util.concurrent.*;

/**
 * 数据源连接测试服务
 */
@Slf4j
@Service
public class DatasourceConnectionService {

    private final ExecutorService executor = Executors.newFixedThreadPool(10);

    /**
     * 测试数据库连接
     */
    public ConnectionTestResult testDatabaseConnection(DatasourceConfig config) {
        long startTime = System.currentTimeMillis();
        
        try {
            // 1. 连接测试
            try (Connection conn = getConnection(config)) {
                boolean valid = conn.isValid(5);
                
                if (!valid) {
                    return ConnectionTestResult.builder()
                        .success(false)
                        .message("连接无效")
                        .duration(System.currentTimeMillis() - startTime)
                        .build();
                }
            }
            
            // 2. 元数据查询
            DatabaseMetaData metaData = getMetaData(config);
            String catalog = metaData.getCatalog();
            String schema = metaData.getSchemaTerm();
            
            // 3. 表列表查询（限制数量）
            List<String> tables = getSampleTables(config, 10);
            
            return ConnectionTestResult.builder()
                .success(true)
                .message("连接成功")
                .catalog(catalog)
                .schema(schema)
                .sampleTables(tables)
                .databaseVersion(metaData.getDatabaseProductVersion())
                .driverVersion(metaData.getDriverVersion())
                .duration(System.currentTimeMillis() - startTime)
                .build();
                
        } catch (Exception e) {
            log.error("连接测试失败: {}", config.getName(), e);
            return ConnectionTestResult.builder()
                .success(false)
                .message("连接失败: " + e.getMessage())
                .errorCode(getErrorCode(e))
                .duration(System.currentTimeMillis() - startTime)
                .build();
        }
    }

    /**
     * 测试Kafka连接
     */
    public ConnectionTestResult testKafkaConnection(DatasourceConfig config) {
        try {
            Properties props = buildKafkaProperties(config);
            
            AdminClient admin = AdminClient.create(props);
            
            // 列出Topics
            ListTopicsResult topics = admin.listTopics();
            Set<String> topicNames = topics.names().get(10, TimeUnit.SECONDS);
            
            admin.close();
            
            return ConnectionTestResult.builder()
                .success(true)
                .message("连接成功")
                .sampleTables(topicNames.stream().limit(10).toList())
                .duration(0L)
                .build();
                
        } catch (Exception e) {
            return ConnectionTestResult.builder()
                .success(false)
                .message("Kafka连接失败: " + e.getMessage())
                .build();
        }
    }

    private String getErrorCode(Exception e) {
        if (e.getMessage() == null) return "UNKNOWN";
        
        if (e.getMessage().contains("Access denied")) return "AUTH_FAILED";
        if (e.getMessage().contains("Unknown database")) return "DB_NOT_FOUND";
        if (e.getMessage().contains("Connection refused")) return "CONN_REFUSED";
        if (e.getMessage().contains("timeout")) return "TIMEOUT";
        
        return "OTHER";
    }
}
```
