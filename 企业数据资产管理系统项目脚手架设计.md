```java
package com.enterprise.edams.common.entity;

import com.baomidou.mybatisplus.annotation.*;
import lombok.Data;
import lombok.NoArgsConstructor;
import lombok.experimental.SuperBuilder;
import java.io.Serializable;
import java.time.LocalDateTime;

/**
 * 基础实体类
 * 包含通用字段：主键、创建人、创建时间、更新人、更新时间、逻辑删除、版本号
 */
@Data
@SuperBuilder
@NoArgsConstructor
public abstract class BaseEntity implements Serializable {

    private static final long serialVersionUID = 1L;

    /**
     * 主键ID
     */
    @TableId(value = "id", type = IdType.AUTO)
    private Long id;

    /**
     * 创建人
     */
    @TableField("created_by")
    @Column(name = "created_by", nullable = false, length = 64)
    private String createdBy;

    /**
     * 创建时间
     */
    @TableField("created_time")
    @Column(name = "created_time", nullable = false)
    private LocalDateTime createdTime;

    /**
     * 更新人
     */
    @TableField("updated_by")
    @Column(name = "updated_by", nullable = false, length = 64)
    private String updatedBy;

    /**
     * 更新时间
     */
    @TableField("updated_time")
    @Column(name = "updated_time", nullable = false)
    private LocalDateTime updatedTime;

    /**
     * 逻辑删除标记
     */
    @TableLogic
    @TableField("deleted")
    @Column(name = "deleted", nullable = false)
    private Integer deleted = 0;

    /**
     * 乐观锁版本
     */
    @Version
    @TableField("version")
    @Column(name = "version")
    private Integer version = 0;

    /**
     * 保存前执行
     */
    @PrePersist
    protected void onCreate() {
        LocalDateTime now = LocalDateTime.now();
        if (this.createdTime == null) {
            this.createdTime = now;
        }
        if (this.updatedTime == null) {
            this.updatedTime = now;
        }
        if (this.createdBy == null || this.createdBy.isEmpty()) {
            this.createdBy = getCurrentUsername();
        }
        if (this.updatedBy == null || this.updatedBy.isEmpty()) {
            this.updatedBy = getCurrentUsername();
        }
    }

    /**
     * 更新前执行
     */
    @PreUpdate
    protected void onUpdate() {
        this.updatedTime = LocalDateTime.now();
        this.updatedBy = getCurrentUsername();
    }

    /**
     * 获取当前用户名
     */
    private String getCurrentUsername() {
        try {
            return SecurityContextHolder.getContext()
                .getAuthentication().getName();
        } catch (Exception e) {
            return "system";
        }
    }
}
```

```java
/**
 * 数据资产管理控制器
 *
 * @author Architecture Team
 * @version 1.0.0
 */
@Slf4j
@RestController
@RequiredArgsConstructor
@RequestMapping("/api/v1/assets")
@Tag(name = "资产管理", description = "数据资产管理相关接口")
public class DataAssetController {

    private final DataAssetService dataAssetService;
    private final AuditService auditService;

    /**
     * 创建数据资产
     */
    @PostMapping
    @Operation(summary = "创建资产", description = "创建新的数据资产记录")
    @PreAuthorize("hasAuthority('asset:create')")
    public Result<DataAssetVO> create(
            @Validated @RequestBody CreateAssetRequest request) {
        
        // 记录审计日志
        auditService.log("CREATE", "data_asset", null, request);
        
        // 创建资产
        DataAsset asset = dataAssetService.create(request);
        
        // 返回结果
        return Result.success(DataAssetConvert.toVO(asset));
    }

    /**
     * 获取资产详情
     */
    @GetMapping("/{id}")
    @Operation(summary = "获取资产详情", description = "根据ID获取数据资产详细信息")
    @PreAuthorize("hasAuthority('asset:read')")
    public Result<DataAssetVO> getById(
            @Parameter(description = "资产ID") @PathVariable Long id) {
        
        DataAsset asset = dataAssetService.getById(id);
        return Result.success(DataAssetConvert.toVO(asset));
    }

    /**
     * 分页查询资产列表
     */
    @GetMapping
    @Operation(summary = "查询资产列表", description = "分页查询数据资产列表")
    @PreAuthorize("hasAuthority('asset:list')")
    public Result<PageResult<DataAssetVO>> list(
            @Parameter(description = "关键词") @RequestParam(required = false) String keyword,
            @Parameter(description = "资产类型") @RequestParam(required = false) String assetType,
            @Parameter(description = "敏感级别") @RequestParam(required = false) String sensitivity,
            @Parameter(description = "状态") @RequestParam(required = false) String status,
            @Parameter(description = "页码") @RequestParam(defaultValue = "1") int pageNum,
            @Parameter(description = "每页数量") @RequestParam(defaultValue = "20") int pageSize) {
        
        // 参数校验
        if (pageSize > 100) {
            pageSize = 100;
        }
        
        // 查询
        Page<DataAsset> page = dataAssetService.search(
            keyword, assetType, sensitivity, status, pageNum, pageSize);
        
        // 转换并返回
        return Result.success(PageResult.of(page, DataAssetConvert::toVO));
    }

    /**
     * 更新资产
     */
    @PutMapping("/{id}")
    @Operation(summary = "更新资产", description = "更新指定ID的数据资产")
    @PreAuthorize("hasAuthority('asset:update')")
    public Result<DataAssetVO> update(

        return Result.success(DataAssetConvert.toVO(asset));
```

```java
package com.enterprise.edams.asset.service.impl;

import com.baomidou.mybatisplus.extension.plugins.pagination.Page;

import com.enterprise.edams.asset.service.DataAssetService;
import com.enterprise.edams.common.exception.BusinessException;
import com.enterprise.edams.common.exception.ErrorCode;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;
import java.util.stream.Collectors;

/**
 * 数据资产服务实现
 */
@Slf4j
@Service
@RequiredArgsConstructor
@Transactional(rollbackFor = Exception.class)
public class DataAssetServiceImpl 
    extends ServiceImpl<DataAssetMapper, DataAsset>
    implements DataAssetService {

    private final DataAssetRepository repository;
    private final DataAssetMapper mapper;

    @Override
    public DataAsset create(CreateAssetRequest request) {
        log.info("创建资产: name={}, type={}", request.getAssetName(), request.getAssetType());
        
        // 1. 参数校验
        validateCreateRequest(request);
        
        // 2. 检查重复
        if (repository.existsByAssetName(request.getAssetName())) {
            throw new BusinessException(
                ErrorCode.RESOURCE_ALREADY_EXISTS,
                String.format("资产名称已存在: %s", request.getAssetName())
            );
        }
        
        // 3. 构建实体
        DataAsset asset = DataAssetConvert.toEntity(request);
        
        // 4. 保存
        DataAsset saved = repository.save(asset);
        
        // 5. 发送事件
        eventPublisher.publish(new AssetCreatedEvent(saved));
        
        log.info("资产创建成功: id={}", saved.getId());
        return saved;
    }

    @Override
    public DataAsset update(Long id, UpdateAssetRequest request) {
        log.info("更新资产: id={}", id);
        
        // 1. 查询资产
        DataAsset asset = repository.findById(id)
            .orElseThrow(() -> new AssetException(id));
        
        // 2. 校验状态
        if (asset.isImmutable()) {
            throw new BusinessException(
                ErrorCode.DATA_INVALID,
                "当前状态不允许修改"
            );
        }
        
        // 3. 更新字段
        DataAssetConvert.updateEntity(asset, request);
        
        // 4. 保存
        DataAsset updated = repository.save(asset);
        
        // 5. 发送事件
        eventPublisher.publish(new AssetUpdatedEvent(updated));
        
        return updated;
    }

    @Override
    public void deleteById(Long id) {
        log.info("删除资产: id={}", id);
        
        // 1. 查询资产
        DataAsset asset = repository.findById(id)
            .orElseThrow(() -> new AssetException(id));
        
        // 2. 检查是否可以删除
        if (asset.isProtected()) {
            throw new BusinessException(
                ErrorCode.DATA_INVALID,
                "受保护资产不允许删除"
            );
        }
        
        // 3. 逻辑删除
        repository.removeById(id);
        
        // 4. 发送事件
        eventPublisher.publish(new AssetDeletedEvent(id));
        
        log.info("资产删除成功: id={}", id);
    }

    @Override
    public DataAsset getById(Long id) {
        return repository.findById(id)
            .orElseThrow(() -> new AssetException(id));
    }

    @Override
    public Page<DataAsset> search(String keyword, String assetType,
                                  String sensitivity, String status,
                                  int pageNum, int pageSize) {
        log.debug("查询资产: keyword={}, type={}, page={}, size={}",
            keyword, assetType, pageNum, pageSize);
        
        // 构建查询条件
        DataAssetQuery query = DataAssetQuery.builder()
            .keyword(keyword)
            .assetType(assetType)
            .sensitivity(sensitivity)
            .status(status)
            .pageNum(pageNum)
            .pageSize(pageSize)
            .build();
        
        return repository.search(query);
    }

    @Override
    public DataAsset publish(Long id) {
        log.info("发布资产: id={}", id);
        
        DataAsset asset = repository.findById(id)
            .orElseThrow(() -> new AssetException(id));
        
        asset.publish();
        
        DataAsset published = repository.save(asset);
        
        eventPublisher.publish(new AssetPublishedEvent(published));
        
        return published;
    }

    @Override
    public int batchDelete(List<Long> ids) {
        log.info("批量删除资产: count={}", ids.size());
        
        int successCount = 0;
        for (Long id : ids) {
            try {
                deleteById(id);
                successCount++;
            } catch (Exception e) {
                log.error("删除资产失败: id={}", id, e);
            }
        }
        
        return successCount;
    }

    @Override
    public boolean existsByName(String name) {
        return repository.existsByAssetName(name);
    }

    // ============ 私有方法 ============

    private void validateCreateRequest(CreateAssetRequest request) {
        if (request == null) {
            throw new IllegalArgumentException("请求参数不能为空");
        }
        if (request.getAssetName() == null || request.getAssetName().trim().isEmpty()) {
            throw new IllegalArgumentException("资产名称不能为空");
        }
        if (request.getAssetName().length() > 200) {
            throw new IllegalArgumentException("资产名称不能超过200个字符");
        }
        if (request.getAssetType() == null) {
            throw new IllegalArgumentException("资产类型不能为空");
        }
    }
}
```

```yaml

  datasource:
    password: AES(${cipher_password})  # 使用AES加密

      password: AES(${cipher_password})
  
  redis:
    password: AES(${cipher_redis_password})

# 解密工具类
@Component
public class ConfigurationDecrypt implements EnvironmentPostProcessor {
    
    @Override
    public void postProcessEnvironment(
            ConfigurableEnvironment environment, 
            SpringApplication application) {
        
        String active = environment.getProperty("spring.profiles.active");
        
        // 加载加密配置
        decryptProperties(environment, "spring.datasource.password");
        decryptProperties(environment, "spring.redis.password");
    }
    
    private void decryptProperties(ConfigurableEnvironment env, String key) {
        String value = env.getProperty(key);
        if (value != null && value.startsWith("AES(")) {
            String cipher = value.substring(5, value.length() - 1);
            String decrypted = AESUtil.decrypt(cipher, secretKey);
            System.setProperty(key, decrypted);
        }
    }
}
```
